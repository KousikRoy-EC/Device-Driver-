The Three Virtual Filesystems — /proc, /sys, /sys/kernel/debug

🗂️ Why Virtual Filesystems?

Rather than inventing a new IPC mechanism for every piece of driver information to expose, Linux reuses its existing filesystem abstraction. Virtual filesystems contain no disk data — reading and writing them calls kernel code directly. Files appear and disappear as drivers load and unload. From userspace they look like ordinary text files, but every cat or echo invokes your driver's callback functions.

📋 /proc

mounted at /proc
  • Process & kernel info
  • Driver stats & state
  • Use: seq_file API
  • Free-form text OK
  • Audience: humans, tools

🔧 sysfs (/sys)

mounted at /sys
  • Device model hierarchy
  • ONE value per file (rule!)
  • show/store callbacks
  • Stable ABI — careful!
  • Audience: udev, scripts

🐛 debugfs

mounted at /sys/kernel/debug
  • Debug & dev only
  • No ABI guarantees
  • Rich built-in types
  • Disabled in production
  • Audience: driver developer
WHERE TO PUT WHAT: ───────────────────────────────────────────────────────────────────────── /proc/driver/mydriver/stats ← aggregate statistics, human-readable /proc/driver/mydriver/status ← current driver state summary /sys/bus/platform/devices/mydev/enable ← single on/off value /sys/bus/platform/devices/mydev/threshold ← single tunable integer /sys/bus/platform/devices/mydev/temperature ← single sensor reading /sys/kernel/debug/mydriver/regs ← raw register dump /sys/kernel/debug/mydriver/verbose ← toggle verbose logging /sys/kernel/debug/mydriver/fifo ← internal buffer state Rule of thumb: Stable ABI that scripts rely on? → sysfs Diagnostic/human-readable/aggregated? → /proc Debug-only, no stability guarantee needed? → debugfs

/proc with seq_file — Multi-Page Kernel Output

📄 Why seq_file? The Buffer Problem

Early drivers used a simple read()-style callback to generate /proc output. The problem: everything had to fit in one PAGE_SIZE buffer. If your driver has thousands of entries, you'd overflow. seq_file iterates your data in chunks, transparently reassembling them for whatever the reader requests — handling buffer sizing, seeking, and partial reads automatically.

Two patterns: single (simple, non-iterated — most drivers use this) and iterator (for large datasets walked element by element).

C — Pattern 1: seq_file single — Simple Proc File
#include <linux/proc_fs.h>    /* proc_create_data, proc_remove */
#include <linux/seq_file.h>   /* seq_file, seq_printf, single_open */

/* The show callback: called once to generate ALL content */
static int mydrv_proc_show(struct seq_file *m, void *v)
{
    struct my_device *dev = m->private;

    /* seq_printf: like printf, writes to the seq_file buffer.
     * Handles all buffer management — just print what you need. */
    seq_printf(m, "Driver:     mydriver v1.0\n");
    seq_printf(m, "Device:     %s\n",   dev->name);
    seq_printf(m, "State:      %s\n",   dev->enabled ? "enabled" : "disabled");
    seq_printf(m, "IRQ:        %d\n",   dev->irq);
    seq_printf(m, "Poll count: %u\n",   dev->poll_count);
    seq_printf(m, "RX bytes:   %llu\n", dev->rx_bytes);
    seq_printf(m, "TX bytes:   %llu\n", dev->tx_bytes);
    seq_printf(m, "Buffer:     %u/%u bytes\n",
               kfifo_len(&dev->fifo), kfifo_size(&dev->fifo));

    /* seq_puts: no formatting, slightly faster than seq_printf */
    seq_puts(m, "\n--- Sensors ---\n");
    struct sensor *s;
    list_for_each_entry(s, &dev->sensor_list, list)
        seq_printf(m, "  [%d] %-16s %3d C  %s\n",
                   s->id, s->name, s->value,
                   s->value > 80 ? "ALERT" : "ok");
    return 0;
}

static int mydrv_proc_open(struct inode *inode, struct file *file)
{
    /* single_open: wraps show() into a seq_file with one iteration.
     * pde_data() retrieves the data pointer set in proc_create_data(). */
    return single_open(file, mydrv_proc_show, pde_data(file_inode(file)));
}

/* Since kernel 5.6, /proc uses proc_ops (not file_operations) */
static const struct proc_ops mydrv_proc_ops = {
    .proc_open    = mydrv_proc_open,
    .proc_read    = seq_read,       /* provided by seq_file */
    .proc_lseek   = seq_lseek,      /* provided by seq_file */
    .proc_release = single_release, /* provided by seq_file */
};

/* In init: */
static struct proc_dir_entry *mydrv_proc_dir;

static int mydrv_proc_init(struct my_device *dev)
{
    /* proc_mkdir: create /proc/driver/mydriver/ directory */
    mydrv_proc_dir = proc_mkdir("driver/mydriver", NULL);
    if (!mydrv_proc_dir) return -ENOMEM;

    /* proc_create_data: name, mode, parent, ops, private_data */
    if (!proc_create_data("status", 0444, mydrv_proc_dir,
                           &mydrv_proc_ops, dev)) {
        proc_remove(mydrv_proc_dir);
        return -ENOMEM;
    }
    return 0;
}

static void mydrv_proc_exit(void)
{
    proc_remove(mydrv_proc_dir); /* recursive: removes dir AND all children */
}
⚠️
proc_ops vs file_operationsSince kernel 5.6, /proc files use struct proc_ops with fields named .proc_open, .proc_read, etc. Using the old file_operations struct for proc entries on modern kernels causes a compile error. For the iterator pattern use seq_release; for single use single_release.
C — Pattern 2: seq_file Iterator — Large Datasets
/* Four callbacks: start, next, stop, show — for paginating large lists */

static void *mydrv_seq_start(struct seq_file *m, loff_t *pos)
{
    struct my_device *dev = m->private;
    /* Return pointer to element at *pos, or NULL if beyond end */
    return idr_get_next(&dev->sensor_idr, (unsigned long *)pos);
}

static void *mydrv_seq_next(struct seq_file *m, void *v, loff_t *pos)
{
    struct my_device *dev = m->private;
    (*pos)++;
    return idr_get_next(&dev->sensor_idr, (unsigned long *)pos);
}

static void  mydrv_seq_stop(struct seq_file *m, void *v) {}
/* stop: release locks/refs held during iteration */

static int mydrv_seq_show(struct seq_file *m, void *v)
{
    struct sensor *s = v;
    seq_printf(m, "%4d  %-16s  %3d C  %s\n",
               s->id, s->name, s->value, s->value > 80 ? "ALERT" : "ok");
    return 0;
}

static const struct seq_operations mydrv_seq_ops = {
    .start = mydrv_seq_start, .next  = mydrv_seq_next,
    .stop  = mydrv_seq_stop,  .show  = mydrv_seq_show,
};

static int mydrv_sensors_open(struct inode *inode, struct file *file)
{
    int ret = seq_open(file, &mydrv_seq_ops);   /* seq_open, not single_open */
    if (ret == 0)
        ((struct seq_file *)file->private_data)->private =
            pde_data(file_inode(file));
    return ret;
}
static const struct proc_ops mydrv_sensors_ops = {
    .proc_open    = mydrv_sensors_open,
    .proc_read    = seq_read, .proc_lseek = seq_lseek,
    .proc_release = seq_release,  /* seq_release (not single_release) for iterator */
};

Writable /proc — Accepting Values from Userspace

C — Writable /proc Entry — echo value > /proc/.../threshold
static ssize_t mydrv_threshold_write(struct file *file,
                                       const char __user *ubuf,
                                       size_t count, loff_t *ppos)
{
    struct my_device *dev = pde_data(file_inode(file));
    char  buf[32]; long val; int ret;

    if (count > sizeof(buf) - 1) return -EINVAL;
    if (copy_from_user(buf, ubuf, count)) return -EFAULT;
    buf[count] = '\0';

    /* kstrtol: safe string-to-long, returns 0 on success.
     * strim() removes leading/trailing whitespace and the trailing newline. */
    ret = kstrtol(strim(buf), 10, &val);
    if (ret) return ret;

    if (val < 0 || val > 100) return -EINVAL;
    dev->threshold = (int)val;
    pr_info("mydrv: threshold set to %d\n", dev->threshold);
    return count;   /* MUST return count, not 0, to avoid infinite retry */
}

static const struct proc_ops mydrv_threshold_ops = {
    .proc_open    = mydrv_proc_open,
    .proc_read    = seq_read,
    .proc_write   = mydrv_threshold_write,   /* add this for writability */
    .proc_lseek   = seq_lseek,
    .proc_release = single_release,
};
/* Create with 0644 so owner can write, everyone can read */
proc_create_data("threshold", 0644, mydrv_proc_dir, &mydrv_threshold_ops, dev);

/* ── kstr* parsing helpers ────────────────────────────────────────── */
/* kstrtol  (str, base, &long)   kstrtoul (str, base, &ulong)         */
/* kstrtoint(str, base, &int)    kstrtou32(str, base, &u32)           */
/* kstrtobool(str, &bool)  → "y","1","on","yes","true" → true         */
/* All return 0 on success, -EINVAL/-ERANGE on parse error.           */

sysfs — The Device Model's Filesystem

🔧 One Value Per File — The Iron Rule

sysfs has one rule: one attribute, one value per file. Every sysfs file holds exactly one value — an integer, a string, a boolean. No multi-column tables. This makes sysfs scriptable: cat /sys/class/thermal/thermal_zone0/temp returns one number that a shell script can read and act on.

sysfs is built on the kernel object model (kobject). Every device, bus, driver, and class has a kobject that automatically appears as a directory in /sys. Your driver gets sysfs entries by being part of the device model.

/sys/ ├── bus/platform/devices/mydevice.0/ ← your platform device │ ├── enable ← 0 or 1 │ ├── threshold ← integer value │ ├── temperature ← read-only sensor reading │ └── driver → ../../drivers/platform/mydriver ├── class/mydev/mydev0/ ← your device class └── kernel/debug/mydriver/ ← debugfs mount point ├── regs └── stats
C — DEVICE_ATTR — The Standard sysfs Pattern
#include <linux/device.h>   /* DEVICE_ATTR_RO, DEVICE_ATTR_RW, sysfs_emit */

/* ── DEVICE_ATTR MACROS ───────────────────────────────────────────────
 * DEVICE_ATTR_RO(name)  → 0444, requires name_show()
 * DEVICE_ATTR_RW(name)  → 0644, requires name_show() AND name_store()
 * DEVICE_ATTR_WO(name)  → 0200, requires name_store()
 * DEVICE_ATTR(name, mode, show_fn, store_fn)  → full control
 * Each macro creates: struct device_attribute dev_attr_##name
 * ──────────────────────────────────────────────────────────────────── */

/* Attribute 1: temperature — read-only sensor reading */
static ssize_t temperature_show(struct device *dev,
                                  struct device_attribute *attr, char *buf)
{
    struct my_device *mydev = dev_get_drvdata(dev);
    /* sysfs_emit: bounds-checked, replaces sprintf().
     * buf is PAGE_SIZE bytes. Value MUST end with '\n'. */
    return sysfs_emit(buf, "%d\n", read_temperature(mydev));
}
DEVICE_ATTR_RO(temperature);   /* creates dev_attr_temperature */

/* Attribute 2: enable — read-write on/off switch */
static ssize_t enable_show(struct device *dev,
                             struct device_attribute *attr, char *buf)
{
    struct my_device *mydev = dev_get_drvdata(dev);
    return sysfs_emit(buf, "%d\n", mydev->enabled ? 1 : 0);
}
static ssize_t enable_store(struct device *dev,
                              struct device_attribute *attr,
                              const char *buf, size_t count)
{
    struct my_device *mydev = dev_get_drvdata(dev);
    bool val;
    if (kstrtobool(buf, &val)) return -EINVAL;
    mydev->enabled = val;
    val ? hardware_enable(mydev) : hardware_disable(mydev);
    return count;   /* ALWAYS return count */
}
DEVICE_ATTR_RW(enable);   /* creates dev_attr_enable */

/* Attribute 3: threshold — read-write integer */
static ssize_t threshold_show(struct device *dev,
                                struct device_attribute *attr, char *buf)
{
    return sysfs_emit(buf, "%u\n", ((struct my_device*)dev_get_drvdata(dev))->threshold);
}
static ssize_t threshold_store(struct device *dev,
                                 struct device_attribute *attr,
                                 const char *buf, size_t count)
{
    struct my_device *mydev = dev_get_drvdata(dev);
    unsigned int val;
    int ret = kstrtouint(buf, 10, &val);
    if (ret) return ret;
    if (val > 100) return -EINVAL;
    mydev->threshold = val;
    return count;
}
DEVICE_ATTR_RW(threshold);

/* ── Group attributes for atomic create/remove ──────────────────── */
static struct attribute *mydev_attrs[] = {
    &dev_attr_temperature.attr,
    &dev_attr_enable.attr,
    &dev_attr_threshold.attr,
    NULL   /* NULL-terminated */
};
ATTRIBUTE_GROUPS(mydev);   /* creates mydev_groups[] */

/* Register/unregister in probe/remove: */
sysfs_create_group(&dev->kobj, &mydev_attr_group);
/* OR pass .groups = mydev_groups to struct device_driver (preferred) */
sysfs_remove_group(&dev->kobj, &mydev_attr_group);
💡
sysfs_emit() replaces sprintf()Since kernel 5.10, always use sysfs_emit(buf, fmt, ...) instead of sprintf(buf, ...) in show callbacks. It's bounds-checked against PAGE_SIZE and handles return values correctly.

Standalone kobject — sysfs Without a Device

C — kobject under /sys/kernel/mydriver/ — No struct device Needed
#include <linux/kobject.h>

static struct kobject *mydrv_kobj;

/* kobj_attribute uses __ATTR macro (different from DEVICE_ATTR) */
static ssize_t verbose_show(struct kobject *kobj,
                              struct kobj_attribute *attr, char *buf)
{
    return sysfs_emit(buf, "%d\n", verbose_mode);
}
static ssize_t verbose_store(struct kobject *kobj,
                               struct kobj_attribute *attr,
                               const char *buf, size_t count)
{
    bool val;
    if (kstrtobool(buf, &val)) return -EINVAL;
    verbose_mode = val;
    return count;
}

static struct kobj_attribute count_attr   = __ATTR_RO(global_count);
static struct kobj_attribute verbose_attr =
    __ATTR(verbose, 0644, verbose_show, verbose_store);

static struct attribute *mydrv_kobj_attrs[] = {
    &count_attr.attr, &verbose_attr.attr, NULL
};
ATTRIBUTE_GROUPS(mydrv_kobj);

static int __init mydrv_sysfs_init(void)
{
    /* Creates /sys/kernel/mydriver/ (kernel_kobj = /sys/kernel) */
    mydrv_kobj = kobject_create_and_add("mydriver", kernel_kobj);
    if (!mydrv_kobj) return -ENOMEM;
    int ret = sysfs_create_group(mydrv_kobj, &mydrv_kobj_attr_group);
    if (ret) kobject_put(mydrv_kobj);
    return ret;
}
static void __exit mydrv_sysfs_exit(void)
{
    kobject_put(mydrv_kobj);  /* dec refcount → removes /sys/kernel/mydriver/ */
}
/* Other kobject parents:  kernel_kobj → /sys/kernel/
 *   power_kobj → /sys/power/    firmware_kobj → /sys/firmware/
 *   fs_kobj    → /sys/fs/       mm_kobj       → /sys/kernel/mm/ */

debugfs — Your Development-Time Swiss Army Knife

🐛 No ABI Contract, Maximum Convenience

debugfs is the right place for anything useful during development but not suitable as a stable interface: raw register dumps, internal queue state, verbose logging toggles, test injection points. It provides helper functions backed by simple variables — no callbacks needed for basic types.

debugfs is disabled by default in production kernels (CONFIG_DEBUG_FS=n). Your driver must handle this with IS_ERR_OR_NULL() checks. Never put critical driver functionality behind debugfs.

C — debugfs — Complete API Reference
#include <linux/debugfs.h>

static struct dentry *mydrv_dbg_dir;

static void mydrv_debugfs_init(struct my_device *dev)
{
    /* Creates /sys/kernel/debug/mydriver/
     * IMPORTANT: Never check individual debugfs_create_* return values.
     * They silently no-op if debugfs is disabled. Only check the dir. */
    mydrv_dbg_dir = debugfs_create_dir("mydriver", NULL);

    /* ── Simple variable files — zero boilerplate ───────────────────
     * Signature: debugfs_create_TYPE(name, mode, parent, &variable)  */
    debugfs_create_u32 ("poll_count",  0444, mydrv_dbg_dir, &dev->poll_count);
    debugfs_create_u32 ("error_count", 0444, mydrv_dbg_dir, &dev->error_count);
    debugfs_create_u64 ("rx_bytes",    0444, mydrv_dbg_dir, &dev->rx_bytes);
    debugfs_create_bool("verbose",     0644, mydrv_dbg_dir, &dev->verbose);
    debugfs_create_x32 ("flags",       0444, mydrv_dbg_dir, &dev->flags);
    /* x32/x64: display in hex — good for bitmask/register values */

    /* Atomic counter — lock-free display */
    debugfs_create_atomic_t("open_count", 0444, mydrv_dbg_dir, &dev->open_count);

    /* Blob: expose a binary buffer as a hex dump */
    static struct debugfs_blob_wrapper fw_blob;
    fw_blob.data = firmware_buf; fw_blob.size = firmware_size;
    debugfs_create_blob("firmware", 0444, mydrv_dbg_dir, &fw_blob);

    /* Custom seq_file: for complex output (register dumps, etc.) */
    debugfs_create_file("regs", 0444, mydrv_dbg_dir, dev, &mydrv_regs_fops);

    /* regset32: dump named hardware registers in one shot */
    static const struct debugfs_reg32 mydrv_regs[] = {
        { "CTRL",   0x00 }, { "STATUS", 0x04 },
        { "DATA",   0x08 }, { "IRQ",    0x0C },
    };
    static struct debugfs_regset32 regset = {
        .regs = mydrv_regs, .nregs = ARRAY_SIZE(mydrv_regs),
        .base = dev->mmio_base,
    };
    debugfs_create_regset32("registers", 0444, mydrv_dbg_dir, &regset);
}

static void mydrv_debugfs_exit(void)
{
    debugfs_remove_recursive(mydrv_dbg_dir);
    /* One call removes the directory AND every file inside it */
}

static int pipdev_proc_show(struct seq_file *m, void *v)
{
    struct pipdev *dev = m->private;
    seq_puts(m, "pipdev Statistics\n=================\n");
    seq_printf(m, "Buffer:     %u / %u bytes used\n",
               kfifo_len(&dev->fifo), kfifo_size(&dev->fifo));

               dev->total_writes, dev->total_tx_bytes);
    seq_printf(m, "Flushes:    %llu\n", dev->flush_count);
    seq_printf(m, "DropOnFull: %s\n", dev->drop_on_full ? "yes" : "no");
    return 0;
}

bash

hello world

1
$ cat /sys/kernel/debug/pipdev/rx_bytes
12

$ echo 1 > /sys/kernel/pipdev/drop_on_full

1

Exercise — Add a writable /proc reset entry

Add /proc/driver/pipdev/reset with a write callback. When the user echoes anything to it, reset all statistics counters (total_reads, total_writes, total_rx_bytes, total_tx_bytes, flush_count) to zero atomically under the mutex. Verify with: write some data, cat stats, then echo 1 > reset, then cat stats again — all counters should be zero.
