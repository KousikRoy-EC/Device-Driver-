The Big Picture — How a Character Device Works

📁 Everything in Linux Is a File

The Unix design philosophy — "everything is a file" — means hardware devices appear as entries in /dev/. A user process does open("/dev/mydevice", O_RDWR) and gets back a file descriptor. Every read(), write(), ioctl() on that fd flows through the VFS (Virtual File System) and into your driver's function pointers defined in struct file_operations.

A character device transfers data character-by-character (stream-based), unlike block devices which transfer in fixed-size blocks. Serial ports, sensors, GPIO controllers, custom hardware — these are all character devices.

USERSPACE KERNEL ───────────────────────────────────────────────────────────────────────── int fd = open("/dev/mydev", O_RDWR) │ ▼ VFS layer ← kernel/fs/open.c looks up /dev/mydev finds it's a char device with major:minor X:Y │ ▼ struct file created ← allocated per open() call file->f_op = &mydev_fops ← YOUR file_operations table your .open() called │ read(fd, buf, 100) ← user calls read ▼ VFS calls file->f_op->read() ▼ your .read() handler ← runs in process context copy_to_user(buf, kbuf, n) ← safe kernel→user copy │ ▼ ioctl(fd, MY_CMD, arg) ← user calls ioctl ▼ your .unlocked_ioctl() │ close(fd) ▼ your .release() called struct file freed
DEVICE NUMBER ANATOMY: dev_t (32-bit value) ───────────────────────────────────────────────────────────────────────── dev_t is a u32: [ MAJOR (12 bits) | MINOR (20 bits) ] MAJOR number → identifies the DRIVER (which driver handles this device) MINOR number → identifies the DEVICE INSTANCE (which device within the driver) Examples from a real system: ls -la /dev/tty* crw-rw-rw- 1 root dialout 4, 64 /dev/ttyS0 ← major=4, minor=64 crw-rw-rw- 1 root dialout 4, 65 /dev/ttyS1 ← major=4, minor=65 crw-rw-rw- 1 root dialout 4, 66 /dev/ttyS2 ← major=4, minor=66 The 'c' prefix means character device ('b' = block device) Macros: MAJOR(dev) → extract major from dev_t MINOR(dev) → extract minor from dev_t MKDEV(major, minor) → compose dev_t from major + minor

Device Number Registration & cdev Setup

🔢 Two-Step Registration Process

Registering a character device is a deliberate two-step process in the kernel. First you claim device numbers (major:minor range). Then you register a struct cdev that binds those numbers to your file_operations. The separation lets you set up data structures between the two steps before any userspace access can reach your driver.

C — Device Registration — The Complete Pattern
#include <linux/fs.h>       /* file_operations, register_chrdev_region */
#include <linux/cdev.h>     /* struct cdev, cdev_init, cdev_add */

/* ── Device state ─────────────────────────────────────────────────── */
struct my_device {
    struct cdev   cdev;        /* embed cdev — connects numbers to fops */
    dev_t         devno;       /* our major:minor */
    /* ... driver-specific fields ... */
};

static struct my_device *mydev;
static struct class    *mydev_class;  /* for auto /dev node creation */
static dev_t            mydev_devno;  /* our device number base */

/* forward declaration — defined in section 4 */
static const struct file_operations mydev_fops;

static int __init mydev_init(void)
{
    int ret;

    mydev = kzalloc(sizeof(*mydev), GFP_KERNEL);
    if (!mydev)
        return -ENOMEM;

    /* ── STEP 1: Claim device numbers ──────────────────────────────
     *
     * Option A — alloc_chrdev_region: kernel assigns a free major number
     *   Use this in new drivers — don't hardcode major numbers.
     *   alloc_chrdev_region(&devno, first_minor, count, name)
     *     devno:       output — the allocated dev_t
     *     first_minor: first minor number you want (usually 0)
     *     count:       how many consecutive minors to reserve
     *     name:        appears in /proc/devices
     */
    ret = alloc_chrdev_region(&mydev_devno, 0, 1, "mydev");
    if (ret < 0) {
        pr_err("mydev: alloc_chrdev_region failed: %d\n", ret);
        goto err_alloc;
    }
    pr_info("mydev: allocated major=%d minor=%d\n",
            MAJOR(mydev_devno), MINOR(mydev_devno));

    /*
     * Option B — register_chrdev_region: you choose the major number
     *   Use only when a fixed major is required (legacy hardware, ABI compat)
     *   register_chrdev_region(MKDEV(major, 0), count, name)
     */
    /* register_chrdev_region(MKDEV(42, 0), 1, "mydev"); */

    /* ── STEP 2: Initialise and add the cdev ───────────────────────
     * cdev_init: links cdev to its file_operations table
     * cdev_add:  makes the device live — after this, opens can arrive! */
    cdev_init(&mydev->cdev, &mydev_fops);
    mydev->cdev.owner = THIS_MODULE;

    ret = cdev_add(&mydev->cdev, mydev_devno, 1);
    if (ret) {
        pr_err("mydev: cdev_add failed: %d\n", ret);
        goto err_cdev;
    }

    /* ── STEP 3: Create /dev/mydev automatically ───────────────────
     * class_create: creates /sys/class/mydev/
     * device_create: creates /sys/class/mydev/mydev0/ and triggers
     *   udev/mdev to create the /dev/mydev node automatically.
     * Without this, the user must manually run mknod. */
    mydev_class = class_create("mydev");
    if (IS_ERR(mydev_class)) {
        ret = PTR_ERR(mydev_class);
        goto err_class;
    }

    if (IS_ERR(device_create(mydev_class, NULL, mydev_devno,
                               NULL, "mydev"))) {
        ret = -ENOMEM;
        goto err_device;
    }

    pr_info("mydev: /dev/mydev created, ready\n");
    return 0;

err_device:
    class_destroy(mydev_class);
err_class:
    cdev_del(&mydev->cdev);
err_cdev:
    unregister_chrdev_region(mydev_devno, 1);
err_alloc:
    kfree(mydev);
    return ret;
}

static void __exit mydev_exit(void)
{
    /* Teardown in REVERSE order of setup */
    device_destroy(mydev_class, mydev_devno);     /* remove /dev/mydev */
    class_destroy(mydev_class);                   /* remove /sys/class/mydev */
    cdev_del(&mydev->cdev);                       /* unlink fops from numbers */
    unregister_chrdev_region(mydev_devno, 1);     /* release major:minor */
    kfree(mydev);
    pr_info("mydev: removed\n");
}
💡
cdev_add is the Point of No ReturnOnce cdev_add() returns successfully, the kernel can immediately route open() calls to your driver. Make sure all your data structures are initialised before calling cdev_add(). In the error path, cdev_del() removes the device atomically.

struct file & struct inode — The Two Key Kernel Objects

📄 struct inode — The Device Node

Represents the /dev/mydev file on disk (in devtmpfs). There is exactly one inode per device node, regardless of how many processes have it open. Your driver uses it during .open() to identify which device (via iminor(inode) / imajor(inode)) and to stash your private data pointer.

  • inode->i_rdev — the dev_t of this device node
  • inode->i_cdev — pointer to the struct cdev
  • imajor(inode) — extract major number
  • iminor(inode) — extract minor number

📂 struct file — The Open File Descriptor

Created fresh for each open() call. One per open file descriptor — two processes opening the same device get separate struct file instances. This is where you store per-open-instance state.

  • file->private_data — your driver's per-open pointer, set in .open(), used in all other ops
  • file->f_flags — O_RDONLY, O_WRONLY, O_NONBLOCK, etc.
  • file->f_pos — current file position (loff_t)
  • file->f_mode — FMODE_READ, FMODE_WRITE

struct file_operations — Your Driver's Function Table

🗂️ The VFS-to-Driver Bridge

struct file_operations is a table of function pointers that the VFS calls when userspace performs operations on your device's file descriptor. You only implement the operations you need — unimplemented fields are NULL, and the VFS returns -EINVAL or a sensible default for those operations.

.owner
THIS_MODULE — prevents module unload while device is open. Always set this.
.open
int (*open)(struct inode *, struct file *) — called on open(). Use to allocate per-open state and set file->private_data. Return 0 on success.
.release
int (*release)(struct inode *, struct file *) — called on close(). Free per-open state. Called once per open, even if multiple file descriptors point to the same open (dup'd fds).
.read
ssize_t (*read)(struct file *, char __user *, size_t, loff_t *) — copy data from kernel to userspace. Use copy_to_user(). Return bytes copied, 0 for EOF, negative errno on error.
.write
ssize_t (*write)(struct file *, const char __user *, size_t, loff_t *) — receive data from userspace. Use copy_from_user(). Return bytes consumed, negative errno on error.
.unlocked_ioctl
long (*unlocked_ioctl)(struct file *, unsigned int cmd, unsigned long arg) — out-of-band control commands. The modern ioctl (no BKL). Use for device-specific configuration.
.poll
__poll_t (*poll)(struct file *, poll_table *) — called by select/poll/epoll. Call poll_wait() and return EPOLLIN/EPOLLOUT masks based on data readiness.
.llseek
loff_t (*llseek)(struct file *, loff_t, int) — handle lseek(). If your device doesn't support seeking, set to no_llseek. Default is default_llseek which updates f_pos.
.mmap
int (*mmap)(struct file *, struct vm_area_struct *) — map device memory into userspace address space. Advanced — covered in Week 3.
.fasync
int (*fasync)(int, struct file *, int) — SIGIO/SIGPOLL async notification support. Calls fasync_helper().
C — file_operations — Canonical Declaration
/* Forward declarations of handler functions (defined further down) */
static int     mydev_open   (struct inode *, struct file *);
static int     mydev_release(struct inode *, struct file *);
static ssize_t mydev_read   (struct file *, char __user *, size_t, loff_t *);
static ssize_t mydev_write  (struct file *, const char __user *, size_t, loff_t *);
static long    mydev_ioctl  (struct file *, unsigned int, unsigned long);

static const struct file_operations mydev_fops = {
    .owner          = THIS_MODULE,
    .open           = mydev_open,
    .release        = mydev_release,
    .read           = mydev_read,
    .write          = mydev_write,
    .unlocked_ioctl = mydev_ioctl,
    .llseek         = no_llseek,    /* this device doesn't support seeking */
};

copy_to_user / copy_from_user — Safe Userspace Transfers

🔒 Why You Can Never Directly Dereference User Pointers

Userspace passes you pointers (char __user *buf). You cannot directly dereference these in kernel code for three reasons: (1) the pointer might be invalid or NULL, (2) the user may not have the right permissions, (3) the user's page might not be in physical memory (page fault needed). copy_to_user() and copy_from_user() handle all of this — they verify access rights, handle page faults, and return the number of bytes that could not be copied.

C — copy_to_user / copy_from_user — Every Variant
#include <linux/uaccess.h>    /* copy_to/from_user, get_user, put_user */

/* ── copy_to_user: kernel buffer → userspace ──────────────────────── */
/* Returns: 0 on success, number of bytes NOT copied on failure.
 * ALWAYS check the return value — partial copies are errors! */

static ssize_t mydev_read(struct file *filp, char __user *ubuf,
                           size_t count, loff_t *ppos)
{
    struct my_device *dev = filp->private_data;
    char  kbuf[256];
    ssize_t to_copy, not_copied;

    /* Clamp to available data and user's requested size */
    to_copy = min((size_t)dev->data_len, count);
    if (to_copy == 0)
        return 0;   /* EOF */

    /* Prepare kernel buffer */
    memcpy(kbuf, dev->data, to_copy);

    not_copied = copy_to_user(ubuf, kbuf, to_copy);
    if (not_copied) {
        pr_err("mydev: copy_to_user failed: %zd bytes not copied\n", not_copied);
        return -EFAULT;  /* bad user pointer */
    }

    /* Advance file position */
    *ppos += to_copy;
    return to_copy;   /* bytes successfully copied */
}

/* ── copy_from_user: userspace → kernel buffer ────────────────────── */
static ssize_t mydev_write(struct file *filp, const char __user *ubuf,
                            size_t count, loff_t *ppos)
{
    struct my_device *dev = filp->private_data;
    char  kbuf[256];
    size_t to_copy = min(count, sizeof(kbuf));
    size_t not_copied;

    not_copied = copy_from_user(kbuf, ubuf, to_copy);
    if (not_copied)
        return -EFAULT;

    /* Process the received data */
    memcpy(dev->data, kbuf, to_copy);
    dev->data_len = to_copy;

    *ppos += to_copy;
    return to_copy;
}

/* ── Single value transfers (more efficient than copy_ for scalars) ── */

/* get_user: read a single value FROM userspace into a kernel variable */
int user_val;
if (get_user(user_val, (int __user *)uarg))   /* uarg is user pointer to int */
    return -EFAULT;
pr_info("Got value %d from user\n", user_val);

/* put_user: write a single kernel value TO userspace */
int kernel_val = 42;
if (put_user(kernel_val, (int __user *)uarg))
    return -EFAULT;

/* ── access_ok: check before a batch of transfers ────────────────── */
/* copy_to/from_user call this internally, but you can call it upfront
 * to avoid checking mid-operation for large transfers: */
if (!access_ok(ubuf, count))
    return -EFAULT;
/* access_ok only checks that the pointer range looks like a valid
 * user address. It does NOT check if the pages are actually mapped. */

/* ── __user annotation ───────────────────────────────────────────── */
/* __user is a sparse annotation — it has no runtime effect.
 * It marks a pointer as pointing to user memory.
 * sparse (static analysis tool) warns if you dereference it directly.
 * Always annotate user pointers with __user in your function signatures. */
🚨
Never Dereference User Pointers DirectlyWriting *ubuf = 'A' or memcpy(dst, ubuf, n) where ubuf is a char __user * is a severe kernel security vulnerability. The user controls that pointer — it could point to kernel memory, causing a privilege escalation. Always use copy_to_user() / copy_from_user(). The kernel's sparse static analyzer will warn if you miss this.

ioctl — Out-of-Band Device Control

⚙️ Device-Specific Commands Beyond Read/Write

ioctl() is the escape hatch for device-specific commands that don't fit the read/write data model — setting baud rate, querying status, triggering a hardware reset, configuring DMA settings, etc. The kernel provides a convention for defining ioctl command numbers that encodes the direction (read/write), data size, and a magic number to avoid collisions.

C — ioctl — Command Definitions & Handler
#include <linux/ioctl.h>

/* ── DEFINING IOCTL COMMANDS ─────────────────────────────────────────
 * Use the _IO / _IOR / _IOW / _IOWR macros. They encode:
 *   magic:  8-bit identifying byte for YOUR driver (pick one not in
 *           Documentation/userspace-api/ioctl/ioctl-number.rst)
 *   nr:     sequential command number within your driver (0, 1, 2...)
 *   type:   C type of the argument (for size calculation)
 *
 * _IO  (magic, nr)         — no argument (just a command)
 * _IOR (magic, nr, type)   — kernel reads FROM user  (user writes to kernel)
 * _IOW (magic, nr, type)   — kernel writes TO user   (user reads from kernel)
 * _IOWR(magic, nr, type)   — bidirectional
 *
 * Naming convention: put these in a shared header used by both driver
 * and userspace application.
 */

#define MYDEV_IOC_MAGIC   'M'   /* unique magic byte for this driver */

/* Command 0: reset the device (no argument) */
#define MYDEV_RESET       _IO  (MYDEV_IOC_MAGIC, 0)

/* Command 1: set configuration — user writes a struct to kernel */
struct mydev_config {
    int  baud_rate;
    int  data_bits;
    bool parity;
};
#define MYDEV_SET_CONFIG  _IOW (MYDEV_IOC_MAGIC, 1, struct mydev_config)

/* Command 2: get current config — kernel writes a struct to user */
#define MYDEV_GET_CONFIG  _IOR (MYDEV_IOC_MAGIC, 2, struct mydev_config)

/* Command 3: get status (a single int) */
#define MYDEV_GET_STATUS  _IOR (MYDEV_IOC_MAGIC, 3, int)

/* Guard: max command number */
#define MYDEV_IOC_MAXNR   3

/* ── IOCTL HANDLER ────────────────────────────────────────────────── */
static long mydev_ioctl(struct file *filp, unsigned int cmd,
                         unsigned long arg)
{
    struct my_device  *dev = filp->private_data;
    struct mydev_config cfg;
    int status;
    int ret = 0;

    /* Verify the command belongs to this driver */
    if (_IOC_TYPE(cmd) != MYDEV_IOC_MAGIC)  return -ENOTTY;
    if (_IOC_NR(cmd) > MYDEV_IOC_MAXNR)      return -ENOTTY;

    /* Check userspace pointer validity for commands that need it */
    if (_IOC_DIR(cmd) & _IOC_READ)   /* kernel will write to user */
        if (!access_ok((void __user *)arg, _IOC_SIZE(cmd)))
            return -EFAULT;
    if (_IOC_DIR(cmd) & _IOC_WRITE)  /* kernel will read from user */
        if (!access_ok((void __user *)arg, _IOC_SIZE(cmd)))
            return -EFAULT;

    switch (cmd) {

    case MYDEV_RESET:
        pr_info("mydev: reset requested by pid %d\n", current->pid);
        hardware_reset(dev);
        break;

    case MYDEV_SET_CONFIG:
        if (copy_from_user(&cfg, (struct mydev_config __user *)arg,
                           sizeof(cfg)))
            return -EFAULT;
        if (cfg.baud_rate < 9600 || cfg.baud_rate > 4000000)
            return -EINVAL;
        dev->config = cfg;
        apply_config(dev);
        pr_info("mydev: config set: baud=%d bits=%d\n",
                cfg.baud_rate, cfg.data_bits);
        break;

    case MYDEV_GET_CONFIG:
        if (copy_to_user((struct mydev_config __user *)arg,
                         &dev->config, sizeof(dev->config)))
            return -EFAULT;
        break;

    case MYDEV_GET_STATUS:
        status = read_hw_status(dev);
        if (put_user(status, (int __user *)arg))
            return -EFAULT;
        break;

    default:
        return -ENOTTY;   /* unknown command — ENOTTY is convention */
    }

    return ret;
}

misc_register — The Simple Path

🪄 One Call Instead of Six — for Simple Devices

For most drivers that only need a single character device node, the misc device API is far simpler. It handles device number allocation, cdev setup, class creation, and device node creation in a single misc_register() call. All misc devices share major number 10, differentiated by their minor number. This is the recommended approach for simple drivers that don't need multiple device nodes.

Real kernel drivers using misc: /dev/random, /dev/urandom, /dev/null, /dev/tpm0, /dev/watchdog.

C — misc_register — 90% Less Boilerplate
#include <linux/miscdevice.h>  /* struct miscdevice, misc_register */
#include <linux/fs.h>

/* Your file_operations are the same as before */
static const struct file_operations mydev_fops = {
    .owner          = THIS_MODULE,
    .open           = mydev_open,
    .release        = mydev_release,
    .read           = mydev_read,
    .write          = mydev_write,
    .unlocked_ioctl = mydev_ioctl,
    .llseek         = no_llseek,
};

/* Define the misc device */
static struct miscdevice mydev_misc = {
    .minor = MISC_DYNAMIC_MINOR,  /* kernel assigns a free minor */
    .name  = "mydev",              /* creates /dev/mydev */
    .fops  = &mydev_fops,
    .mode  = 0666,                 /* optional: default permissions */
};

static int __init mydev_init(void)
{
    int ret;

    /* Allocate driver state */
    mydev = kzalloc(sizeof(*mydev), GFP_KERNEL);
    if (!mydev)
        return -ENOMEM;

    /* One call replaces: alloc_chrdev_region + cdev_init + cdev_add
     *                    + class_create + device_create             */
    ret = misc_register(&mydev_misc);
    if (ret) {
        pr_err("mydev: misc_register failed: %d\n", ret);
        kfree(mydev);
        return ret;
    }

    pr_info("mydev: registered as /dev/mydev (minor=%d)\n",
            mydev_misc.minor);
    return 0;
}

static void __exit mydev_exit(void)
{
    misc_deregister(&mydev_misc);    /* one call tears it all down */
    kfree(mydev);
    pr_info("mydev: deregistered\n");
}

alloc_chrdev_region + cdev_add Multiple device instances (e.g., /dev/mydev0, /dev/mydev1...), full control needed Medium — 6 calls Yes — reserve a range of minors

Day 8 Practice Project — "Pipe" Character Device

🎯
Today's GoalBuild a complete character device driver called pipdev — a kernel-space circular buffer exposed as /dev/pipdev. One process writes data in, another reads it out. Uses misc_register, a kfifo as the backing store, a mutex for thread safety, a wait_queue for blocking reads, and a full ioctl for buffer management. This is the canonical Day 8 driver — everything you need for Week 2 and beyond.
C — pipdev.c — Complete Character Device Driver
/*
 * pipdev.c — Day 8: Character Device Driver
 *
 * A kernel-space pipe: write bytes in, read bytes out.
 * Backed by a kfifo. Supports blocking reads and ioctl.
 *
 * Build: make
 * Load:  sudo insmod pipdev.ko
 * Use:   echo "hello kernel" > /dev/pipdev
 *        cat /dev/pipdev
 *        sudo insmod pipdev.ko buf_size=8192
 */
#include <linux/init.h>
#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/fs.h>
#include <linux/miscdevice.h>
#include <linux/slab.h>
#include <linux/uaccess.h>
#include <linux/kfifo.h>
#include <linux/mutex.h>
#include <linux/wait.h>
#include <linux/poll.h>
#include <linux/ioctl.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Your Name");
MODULE_DESCRIPTION("Day 8: pipdev - character device pipe driver");

/* ── Module parameters ─────────────────────────────────────────────── */
static int buf_size = 4096;
module_param(buf_size, int, 0444);
MODULE_PARM_DESC(buf_size, "Pipe buffer size in bytes (default: 4096, must be power-of-2)");

/* ── ioctl commands ────────────────────────────────────────────────── */
#define PIPDEV_IOC_MAGIC  'P'
#define PIPDEV_FLUSH      _IO  (PIPDEV_IOC_MAGIC, 0)          /* clear buffer */
#define PIPDEV_GET_AVAIL  _IOR (PIPDEV_IOC_MAGIC, 1, int)   /* bytes available to read */
#define PIPDEV_GET_FREE   _IOR (PIPDEV_IOC_MAGIC, 2, int)   /* free space in buffer */
#define PIPDEV_IOC_MAXNR  2

/* ── Device state ──────────────────────────────────────────────────── */
struct pipdev {
    struct kfifo        fifo;         /* circular buffer */
    struct mutex        lock;         /* protects fifo access */
    wait_queue_head_t   read_wq;      /* readers wait here */
    wait_queue_head_t   write_wq;     /* writers wait here (for space) */
    atomic_t            open_count;   /* number of current opens */
};

static struct pipdev *pd;   /* global device instance */

/* ── open ─────────────────────────────────────────────────────────── */
static int pipdev_open(struct inode *inode, struct file *filp)
{
    pr_debug("pipdev: open by pid=%d flags=0x%x\n",
             current->pid, filp->f_flags);

    /* Store device pointer for use in read/write/ioctl */
    filp->private_data = pd;
    atomic_inc(&pd->open_count);

    return nonseekable_open(inode, filp);  /* mark as non-seekable */
}

/* ── release ──────────────────────────────────────────────────────── */
static int pipdev_release(struct inode *inode, struct file *filp)
{
    atomic_dec(&pd->open_count);
    pr_debug("pipdev: closed by pid=%d, open_count=%d\n",
             current->pid, atomic_read(&pd->open_count));
    return 0;
}

/* ── read ─────────────────────────────────────────────────────────── */
static ssize_t pipdev_read(struct file *filp, char __user *ubuf,
                            size_t count, loff_t *ppos)
{
    struct pipdev *dev = filp->private_data;
    unsigned int copied;
    int ret;

    if (count == 0)
        return 0;

    /* Non-blocking: return immediately if no data */
    if (filp->f_flags & O_NONBLOCK) {
        if (kfifo_is_empty(&dev->fifo))
            return -EAGAIN;
    } else {
        /* Blocking read: sleep until data is available */
        ret = wait_event_interruptible(dev->read_wq,
                                       !kfifo_is_empty(&dev->fifo));
        if (ret)
            return -EINTR;
    }

    /* Acquire lock and drain kfifo to userspace */
    if (mutex_lock_interruptible(&dev->lock))
        return -EINTR;

    ret = kfifo_to_user(&dev->fifo, ubuf, count, &copied);

    mutex_unlock(&dev->lock);

    if (ret)
        return -EFAULT;

    /* Wake any waiting writers — there's space now */
    if (copied > 0)
        wake_up_interruptible(&dev->write_wq);

    pr_debug("pipdev: read %u bytes\n", copied);
    return copied;
}

/* ── write ────────────────────────────────────────────────────────── */
static ssize_t pipdev_write(struct file *filp, const char __user *ubuf,
                             size_t count, loff_t *ppos)
{
    struct pipdev *dev = filp->private_data;
    unsigned int copied;
    int ret;

    if (count == 0)
        return 0;

    /* Non-blocking: fail immediately if buffer full */
    if (filp->f_flags & O_NONBLOCK) {
        if (kfifo_is_full(&dev->fifo))
            return -EAGAIN;
    } else {
        /* Blocking write: sleep until space is available */
        ret = wait_event_interruptible(dev->write_wq,
                                       !kfifo_is_full(&dev->fifo));
        if (ret)
            return -EINTR;
    }

    if (mutex_lock_interruptible(&dev->lock))
        return -EINTR;

    ret = kfifo_from_user(&dev->fifo, ubuf, count, &copied);

    mutex_unlock(&dev->lock);

    if (ret)
        return -EFAULT;

    /* Wake any readers — data is now available */
    if (copied > 0)
        wake_up_interruptible(&dev->read_wq);

    pr_debug("pipdev: wrote %u bytes (%u avail)\n",
             copied, kfifo_len(&dev->fifo));
    return copied;
}

/* ── poll ─────────────────────────────────────────────────────────── */
static __poll_t pipdev_poll(struct file *filp, poll_table *wait)
{
    struct pipdev *dev = filp->private_data;
    __poll_t mask = 0;

    /* Register with both wait queues — poll will call us again when either fires */
    poll_wait(filp, &dev->read_wq,  wait);
    poll_wait(filp, &dev->write_wq, wait);

    if (!kfifo_is_empty(&dev->fifo))
        mask |= EPOLLIN | EPOLLRDNORM;   /* data available to read */
    if (!kfifo_is_full(&dev->fifo))
        mask |= EPOLLOUT | EPOLLWRNORM;  /* space available to write */

    return mask;
}

/* ── ioctl ────────────────────────────────────────────────────────── */
static long pipdev_ioctl(struct file *filp, unsigned int cmd,
                          unsigned long arg)
{
    struct pipdev *dev = filp->private_data;
    int val;

    if (_IOC_TYPE(cmd) != PIPDEV_IOC_MAGIC)  return -ENOTTY;
    if (_IOC_NR(cmd) > PIPDEV_IOC_MAXNR)      return -ENOTTY;

    switch (cmd) {
    case PIPDEV_FLUSH:
        mutex_lock(&dev->lock);
        kfifo_reset(&dev->fifo);
        mutex_unlock(&dev->lock);
        wake_up_interruptible(&dev->write_wq);
        pr_info("pipdev: buffer flushed by pid=%d\n", current->pid);
        break;

    case PIPDEV_GET_AVAIL:
        val = kfifo_len(&dev->fifo);
        if (put_user(val, (int __user *)arg))
            return -EFAULT;
        break;

    case PIPDEV_GET_FREE:
        val = kfifo_avail(&dev->fifo);
        if (put_user(val, (int __user *)arg))
            return -EFAULT;
        break;

    default:
        return -ENOTTY;
    }
    return 0;
}

static const struct file_operations pipdev_fops = {
    .owner          = THIS_MODULE,
    .open           = pipdev_open,
    .release        = pipdev_release,
    .read           = pipdev_read,
    .write          = pipdev_write,
    .poll           = pipdev_poll,
    .unlocked_ioctl = pipdev_ioctl,
    .llseek         = no_llseek,
};

static struct miscdevice pipdev_misc = {
    .minor = MISC_DYNAMIC_MINOR,
    .name  = "pipdev",
    .fops  = &pipdev_fops,
    .mode  = 0666,
};

static int __init pipdev_init(void)
{
    int ret;

    if (!is_power_of_2(buf_size)) {
        pr_err("pipdev: buf_size must be a power of 2\n");
        return -EINVAL;
    }

    pd = kzalloc(sizeof(*pd), GFP_KERNEL);
    if (!pd)
        return -ENOMEM;

    ret = kfifo_alloc(&pd->fifo, buf_size, GFP_KERNEL);
    if (ret)
        goto err_fifo;

    mutex_init(&pd->lock);
    init_waitqueue_head(&pd->read_wq);
    init_waitqueue_head(&pd->write_wq);
    atomic_set(&pd->open_count, 0);

    ret = misc_register(&pipdev_misc);
    if (ret)
        goto err_misc;

    pr_info("pipdev: /dev/pipdev ready (buf=%d bytes, minor=%d)\n",
            buf_size, pipdev_misc.minor);
    return 0;

err_misc:
    kfifo_free(&pd->fifo);
err_fifo:
    kfree(pd);
    return ret;
}

static void __exit pipdev_exit(void)
{
    misc_deregister(&pipdev_misc);
    kfifo_free(&pd->fifo);
    kfree(pd);
    pr_info("pipdev: removed\n");
}

module_init(pipdev_init);
module_exit(pipdev_exit);

Userspace C test program — test_pipdev.c

C — test_pipdev.c (compile and run in userspace)
#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <string.h>
#include <sys/ioctl.h>

/* Must match the definitions in pipdev.c */
#define PIPDEV_IOC_MAGIC  'P'
#define PIPDEV_FLUSH      _IO  (PIPDEV_IOC_MAGIC, 0)
#define PIPDEV_GET_AVAIL  _IOR (PIPDEV_IOC_MAGIC, 1, int)
#define PIPDEV_GET_FREE   _IOR (PIPDEV_IOC_MAGIC, 2, int)

int main(void)
{
    int fd, avail, free_space;
    char buf[64];

    fd = open("/dev/pipdev", O_RDWR);
    if (fd < 0) { perror("open"); return 1; }

    /* Query free space before writing */
    ioctl(fd, PIPDEV_GET_FREE, &free_space);
    printf("Free before write: %d bytes\n", free_space);

    /* Write a message */
    const char *msg = "Hello from userspace!";
    write(fd, msg, strlen(msg));

    /* Query available data */
    ioctl(fd, PIPDEV_GET_AVAIL, &avail);
    printf("Available after write: %d bytes\n", avail);

    /* Read it back */
    int n = read(fd, buf, sizeof(buf) - 1);
    buf[n] = '\0';
    printf("Read: '%s'\n", buf);

    /* Flush the buffer */
    ioctl(fd, PIPDEV_FLUSH);
    ioctl(fd, PIPDEV_GET_AVAIL, &avail);
    printf("Available after flush: %d bytes\n", avail);

    close(fd);
    return 0;
}
