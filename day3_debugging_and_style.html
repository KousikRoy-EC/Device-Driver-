The Kernel Ring Buffer — How printk Really Works

🔄 What is the Ring Buffer?

When your driver calls printk(), the message is written to a fixed-size circular buffer in kernel memory called the kernel ring buffer (also called the log buffer). This buffer is defined by CONFIG_LOG_BUF_SHIFT — default is 17 (2^17 = 128KB), but production systems often use 18 or 19 (256KB / 512KB).

The buffer is circular — when it fills up, old messages are overwritten. This means on busy systems, early boot messages can be lost by the time you read them. dmesg is just reading this buffer from /dev/kmsg.

Dynamic Debug — The Right Way to Debug Drivers

🎚️ What is Dynamic Debug?

pr_debug() and dev_dbg() calls are compiled into the kernel but disabled by default. They have zero runtime cost when disabled. The dynamic_debug infrastructure lets you selectively enable them at runtime — per module, per file, per function, even per line — without recompiling.

This is the professional way to add debug output to drivers. You never remove debug calls — you leave them in and control them via dynamic debug.

💡
RequirementYour kernel must have CONFIG_DYNAMIC_DEBUG=y (enabled in most distro kernels). Check: grep DYNAMIC_DEBUG /boot/config-$(uname -r)
bash — Dynamic Debug Control Interface
# The control file — every pr_debug/dev_dbg call is an entry here
sudo cat /sys/kernel/debug/dynamic_debug/control | head -20
# Format: filename:lineno [module]function flags format
# e1000/e1000_main.c:2341 [e1000]e1000_probe =_ "e1000: %s NIC Link is Up\012"
# flags: p=print, f=show function, l=show line, m=show module, t=show thread

# ── ENABLE DEBUG MESSAGES ──────────────────────────────────────────

# Enable ALL debug in your module
sudo sh -c 'echo "module hello +p" > /sys/kernel/debug/dynamic_debug/control'

# Enable debug for a specific file only
sudo sh -c 'echo "file hello.c +p" > /sys/kernel/debug/dynamic_debug/control'

# Enable debug for a specific function
sudo sh -c 'echo "func hello_init +p" > /sys/kernel/debug/dynamic_debug/control'

# Enable debug for a specific line number
sudo sh -c 'echo "file hello.c line 42 +p" > /sys/kernel/debug/dynamic_debug/control'

# Enable with EXTRA info: function name (f), line number (l), module (m)
sudo sh -c 'echo "module hello +pflm" > /sys/kernel/debug/dynamic_debug/control'
# Output becomes: [  892.1] hello hello_init:42 Debug message here

# ── DISABLE DEBUG MESSAGES ────────────────────────────────────────
sudo sh -c 'echo "module hello -p" > /sys/kernel/debug/dynamic_debug/control'

# ── ENABLE AT BOOT (kernel command line) ──────────────────────────
# Add to GRUB_CMDLINE_LINUX in /etc/default/grub:
# dyndbg="module hello +p"

# ── ENABLE AT MODULE LOAD ─────────────────────────────────────────
sudo insmod hello.ko dyndbg="+p"     # enable all debug for this module
sudo modprobe hello dyndbg="+pflm"   # with function/line/module info
C — How to Use pr_debug Correctly in Your Driver
#include <linux/module.h>
#include <linux/kernel.h>

/* IMPORTANT: If you want pr_debug() to be dynamic (not compiled out entirely),
 * you must NOT define DEBUG. With DEBUG defined, pr_debug becomes a regular
 * printk(KERN_DEBUG) always compiled in.
 * Without DEBUG: pr_debug uses dynamic debug infrastructure → zero cost when off */

static int process_data(int *buf, int len)
{
    int i, sum = 0;

    pr_debug("process_data called: buf=%p, len=%d\n", buf, len);  /* debug entry */

    for (i = 0; i < len; i++) {
        sum += buf[i];
        pr_debug("  buf[%d] = %d, running sum = %d\n", i, buf[i], sum);
    }

    pr_debug("process_data done: sum=%d\n", sum);               /* debug exit */
    return sum;
}

/* In a driver with struct device *dev, always prefer dev_dbg(): */
static int sensor_read(struct device *dev, int *value)
{
    dev_dbg(dev, "Reading sensor register 0x%02x\n", REG_DATA);
    /* dev_dbg prepends the device name automatically:
     * [  1.234] sensor 1-0048: Reading sensor register 0x00
     * This tells you WHICH device instance caused the message — crucial
     * when you have multiple devices using the same driver */
    return 0;
}

BUG(), BUG_ON(), WARN(), WARN_ON() — When to Use Each

💣 The Assertion Macros — Your Safety Net

These macros let you assert invariants in your code. When an invariant is violated, they trigger controlled kernel error reporting. Choosing the right one is critical — using BUG() where WARN() is appropriate will cause unnecessary system crashes during development.

MacroTriggers WhenWhat HappensSystem Survives?Use When
BUG() Always (unconditional) Prints oops with full stack trace, kills current process, may panic Maybe — if panic_on_oops=0 Reached an impossible code path. State is completely unrecoverable. E.g., default: in a switch that should never happen.
BUG_ON(cond) When cond is true Same as BUG() if condition is true Maybe Assert a condition that must NEVER be true: BUG_ON(ptr == NULL), BUG_ON(count < 0)
WARN(cond, fmt, ...) When cond is true Prints one-time warning with stack trace to dmesg. Execution continues. Yes — always Something unexpected but recoverable. Driver can continue safely.
WARN_ON(cond) When cond is true Prints one-time warning + stack trace. Returns the condition value. Yes — always Most common for driver debugging. Used as: if (WARN_ON(ret < 0)) return ret;
WARN_ONCE(cond, fmt) First time cond is true Same as WARN but fires exactly once per boot Yes Warning about deprecated usage or one-time events that shouldn't spam dmesg
might_sleep() In atomic context WARN if called where sleeping is not allowed (spinlock held, interrupt context) Yes Add at top of any function that might sleep — helps catch bugs during development
C — Correct Usage of All Assert Macros
#include <linux/bug.h>      /* BUG, BUG_ON, WARN, WARN_ON */
#include <linux/kernel.h>   /* might_sleep */

enum device_state {
    DEV_IDLE = 0,
    DEV_RUNNING,
    DEV_ERROR,
    DEV_STATE_MAX   /* sentinel */
};

static void handle_state(enum device_state state, struct my_dev *dev)
{
    /* BUG_ON: This pointer must NEVER be NULL at this point.
     * If it is, our caller has a serious bug — there's no safe recovery. */
    BUG_ON(dev == NULL);

    switch (state) {
    case DEV_IDLE:
        pr_info("Device is idle\n");
        break;

    case DEV_RUNNING:
        pr_info("Device is running\n");
        break;

    case DEV_ERROR:
        /* WARN: unexpected but we can handle it */
        WARN(1, "Device entered error state unexpectedly\n");
        reset_device(dev);  /* try to recover */
        break;

    default:
        /* BUG: this switch is exhaustive. Reaching default means
         * someone added a state without updating this function — 
         * that's a programming bug, not a runtime error. */
        BUG();
    }
}

static int read_register(struct my_dev *dev, u32 reg, u32 *val)
{
    int ret;

    /* might_sleep: assert this function is only called in process context */
    might_sleep();

    ret = do_i2c_read(dev, reg, val);

    /* WARN_ON: if ret < 0, print a warning with stack trace and
     * return the error to caller. The 'if' makes this a useful pattern. */
    if (WARN_ON(ret < 0)) {
        pr_err("Failed to read register 0x%x: %d\n", reg, ret);
        return ret;
    }

    /* WARN_ONCE: device responded slowly — noteworthy but not actionable */
    WARN_ONCE(dev->latency_ms > 100,
        "Device responding slowly: %dms (expected < 100ms)\n",
        dev->latency_ms);

    return 0;
}
⚠️
Golden Rule: BUG() vs WARN()Ask yourself: "If this condition is true, can the kernel continue safely?" If YES → use WARN(). If NO (state is completely corrupt, continuing will cause data loss or worse oopses) → use BUG_ON(). In practice, prefer WARN_ON() — let the system survive and collect more debug info. Reserve BUG_ON() for true invariant violations where continuing is dangerous.

Reading a Kernel Oops — The Most Important Debugging Skill

💥 What is a Kernel Oops?

An oops is what the kernel prints when it detects an unrecoverable error in kernel code — most commonly a NULL pointer dereference, use-after-free, or stack overflow. Unlike userspace segfaults (which just kill the process), an oops can kill the entire system if panic_on_oops=1.

Reading oops output is the #1 skill that separates experienced kernel developers from beginners. Here is a real oops, annotated line by line.

🔴 Kernel Oops Output — NULL Pointer Dereference in a Driver
BUG: unable to handle kernel NULL pointer dereference at 0000000000000010
   ↑ Type of fault: tried to access virtual address 0x10 (= NULL + offset 0x10)
     This tells us we have a valid struct pointer, but a FIELD inside it is NULL

PF flags: [WRITE] [USER]
   ↑ Page fault flags: WRITE = tried to write. USER = happened in user context
     [PRESENT] missing = page not present (NULL). [WRITE] = write access.

RIP: 0010:hello_write+0x2a/0x80 [hello]
   ↑ RIP = Instruction Pointer (where the crash happened)
     hello_write = function name (from kallsyms)
     +0x2a = offset INTO that function (0x2a bytes from function start)
     /0x80 = total function size (0x80 = 128 bytes)
     [hello] = which module. If blank, it's core kernel code.

Code: 48 8b 47 10 48 89 04 24 e8 ...
   ↑ Raw x86 bytes of the crashing instruction. Useful with objdump.

RSP: 0018:ffffc9000040bde8  EFLAGS: 00010246
RAX: 0000000000000000  RBX: ffff888100b4e800  RCX: 0000000000000005
RDX: 0000000000000005  RSI: 00007ffd8f3a2b20  RDI: ffff888100b4e800
   ↑ CPU register dump at crash time. RAX=0x0 often confirms a NULL pointer.
     RSI = 0x00007ffd... = a USERSPACE address (note the low bits)
     RBX/RDI = 0xffff888... = kernel addresses

CR2: 0000000000000010
   ↑ CR2 = the EXACT address that caused the page fault = 0x10 = NULL+0x10
     This is the most important register for NULL deref debugging

Call Trace:
   ↑ Stack trace — read BOTTOM to TOP for call order:
 <TASK>

 __x64_sys_write+0x19/0x30       ← x86_64 syscall entry

 </TASK>

Modules linked in: hello(OE) e1000 video
   ↑ All loaded modules at crash time. O=out-of-tree, E=unsigned (tainted)

Understand What CR2 = 0x10 Tells You

The faulting address 0x10 = 16 in decimal. This means the code accessed a pointer at offset 16 from a NULL base pointer. Look at your struct — which field is at byte offset 16?

C — Finding the Struct Field at Offset 0x10
struct my_device {
    int   id;           /* offset 0x00, size 4 */
    int   flags;        /* offset 0x04, size 4 */
    void *data;         /* offset 0x08, size 8 (64-bit pointer) */
    char *name;         /* offset 0x10, size 8  ← this is what we accessed! */
};

/* To check offsets programmatically: */
pr_info("name offset: %zu\n", offsetof(struct my_device, name));
/* or use pahole tool: pahole hello.ko */

likely() and unlikely() — Branch Prediction Hints

⚡ What They Are and Why They Matter

Modern CPUs use branch prediction to speculatively execute code before a branch is resolved. If the prediction is wrong, the CPU flushes the pipeline — expensive. likely() and unlikely() are compiler hints that tell GCC which branch is expected to be taken, allowing it to optimize the generated assembly for the common case.

C — Without Hints (baseline)
int do_work(struct buffer *buf)
{
    if (buf == NULL)   /* rarely NULL */
        return -EINVAL;

    /* ... normal processing ... */
    return 0;
}
/* Compiler has no idea which branch is
   more likely → generates "balanced" code
   → extra pipeline flush cost on every
   NULL check even when buf is valid */
C — With Hints (optimized)
int do_work(struct buffer *buf)
{
    if (unlikely(buf == NULL)) /* hint: rare */
        return -EINVAL;

    /* ... normal processing ... */
    return 0;
}
/* Compiler puts error path code FAR from
   the main code path → I-cache stays hot
   with normal code → faster hot path */
C — likely/unlikely Patterns in Real Driver Code
#include <linux/compiler.h>  /* defines likely() and unlikely() */

/* ── UNLIKELY: error conditions are rare ─────────────────────────── */

/* kmalloc fails rarely — hint: unlikely to fail */
buf = kmalloc(size, GFP_KERNEL);
if (unlikely(!buf)) {
    pr_err("Out of memory\n");
    return -ENOMEM;
}

/* Hardware errors are rare during normal operation */
status = read_hw_status(dev);
if (unlikely(status & HW_ERROR_BIT)) {
    handle_error(dev);
    return -EIO;
}

/* ── LIKELY: normal/expected conditions ──────────────────────────── */

/* Device is almost always ready */
if (likely(dev->state == DEV_READY)) {
    process_data(dev);
} else {
    wait_for_device(dev);
}

/* ── IMPLEMENTATION (how they actually work) ──────────────────────── */
/* In include/linux/compiler.h: */
/*   #define likely(x)   __builtin_expect(!!(x), 1) */
/*   #define unlikely(x) __builtin_expect(!!(x), 0) */
/* __builtin_expect is a GCC intrinsic that provides branch prediction hints.
 * The !! converts the value to 0 or 1 (handles non-boolean values). */

/* ── WHEN NOT TO USE THEM ────────────────────────────────────────── */
/* DON'T use on branches where both paths are equally common:
 *   if (likely(x > 0))   ← wrong if x is sometimes > 0 and sometimes ≤ 0
 * DON'T use just for style — only when you KNOW the probability
 * The kernel style guide says: use only when the result impacts performance
 *   (hot paths: interrupt handlers, packet processing, scheduler) */

Kernel Coding Style — Rules Every Driver Must Follow

📋
Why This Matters for Your JobEvery patch submitted to the Linux kernel is reviewed against the coding style. Reviewers will reject or delay patches that violate it. In your job, your driver code will be reviewed by colleagues who expect this style. Learn it now — it becomes muscle memory.

📐 The Rules — Directly From Documentation/process/coding-style.rst

❌ WRONG — Will Be Rejected
// Wrong indentation: spaces
int myFunction(int x) {
    if (x > 0) {
        return x;
    }
    else {
        return -x;
    }
}

// Wrong brace style
if (condition)
{
    do_something();
}

// Wrong spacing around operators
int y=x+1;
if(y>0)
    return(y);

// Wrong: typedef hides struct
typedef struct {
    int val;
} myStruct_t;

// Wrong: camelCase
void doSomeWork() { }
✅ CORRECT — Kernel Style
/* Correct: TAB indentation */
int my_function(int x)
{
    if (x > 0)
        return x;   /* no braces for single stmt */
    return -x;
}

/* Correct: K&R brace style for functions,
 * Allman style for control flow */
if (condition) {   /* brace on same line */
    do_something();
    do_more();     /* braces when >1 stmt */
}

/* Correct: spaces around operators */
int y = x + 1;
if (y > 0)
    return y;     /* no parens around return */

/* Correct: no typedef for structs */
struct my_device {
    int val;
};

/* Correct: snake_case */
void do_some_work(void) { }
RuleCorrectWrongReason
Indentation TAB characters (8-space equivalent visual width) Spaces, 2-space, 4-space tabs 8-space indentation intentionally makes deeply nested code look ugly — encourages refactoring into functions
Line length 80 columns max (hard rule for core, 100 tolerated in some subsystems) Lines > 80-100 chars Readability, terminal compatibility, side-by-side diffs
Naming snake_case for everything: variables, functions, structs, files camelCase, PascalCase, Hungarian notation C global convention; camelCase is C++ style
Braces — functions Opening brace on next line: int foo()\n{ int foo() { K&R C style for function definitions
Braces — control flow Opening brace on same line: if (x) { if (x)\n{ Saves vertical space in control structures
Single-statement if No braces: if (x)\n\treturn 0; if (x) { return 0; } Cleaner, unless there's an else clause with braces
Comments /* C-style comments */ // C++ style Kernel was originally C89. C++ comments now tolerated but C-style preferred
Typedefs Never typedef structs. Use struct my_dev *dev typedef struct {} my_dev_t Typedefs hide the type. In kernel code, knowing something is a struct matters.
goto Allowed for error cleanup only — the "goto cleanup" pattern Using goto for anything else The goto cleanup pattern is cleaner than deeply nested ifs for resource teardown
C — The goto Cleanup Pattern — Canonical Kernel Error Handling
/* This is THE standard pattern for resource management in kernel init functions.
 * It is not a code smell — it is idiomatic kernel C. Learn it and use it. */

static int __init my_driver_init(void)
{
    int ret;

    /* Step 1: Allocate memory */
    my_dev = kzalloc(sizeof(*my_dev), GFP_KERNEL);
    if (!my_dev) {
        ret = -ENOMEM;
        goto err_alloc;           /* jump to cleanup */
    }

    /* Step 2: Register device */
    ret = register_chrdev_region(...);
    if (ret < 0) {
        pr_err("Failed to register: %d\n", ret);
        goto err_register;        /* kfree my_dev, then return */
    }

    /* Step 3: Request IRQ */
    ret = request_irq(irq, my_handler, 0, "mydev", my_dev);
    if (ret < 0) {
        pr_err("Failed to get IRQ: %d\n", ret);
        goto err_irq;             /* unregister, kfree, return */
    }

    pr_info("Driver loaded OK\n");
    return 0;                     /* SUCCESS: skip all cleanup labels */

/* ── ERROR CLEANUP — in REVERSE order of acquisition ─────────────── */
err_irq:
    unregister_chrdev_region(...);
err_register:
    kfree(my_dev);
err_alloc:
    return ret;
}

/* KEY RULES for this pattern:
 * 1. Cleanup labels are in REVERSE order of acquisition
 * 2. Each label only undoes what the CORRESPONDING step did
 * 3. Early labels fall through to later ones (no return between labels)
 * 4. Label names are descriptive: err_alloc, err_register, err_irq
 * This pattern is impossible to get wrong once you understand the "waterfall" */

/proc/kallsyms Deep Dive — The Kernel Symbol Map

🗺️ Every Kernel Symbol — Address, Type, Module

/proc/kallsyms maps every symbol in the running kernel to its virtual address. It includes core kernel symbols, all loaded module symbols, and percpu variables. As a driver developer, you use it to decode crash addresses, understand memory layout, and verify your module's symbols are loaded correctly.

bash — Working with /proc/kallsyms
# Basic format: address  type  name  [module]
$ sudo cat /proc/kallsyms | head -5
0000000000000000 A fixed_percpu_data
0000000000000000 A __per_cpu_start
ffffffff81000000 T startup_64
ffffffff81000000 T _stext
ffffffff81001000 T do_syscall_64

# Symbol type letters (from nm man page):
# T/t = code (.text): T=global, t=local/static
# D/d = initialized data (.data)
# B/b = uninitialized data (.bss)
# R/r = read-only data (.rodata)
# A   = absolute (address doesn't change with KASLR)
# W/w = weak symbol

# Find all symbols from your module
$ sudo grep "\[hello\]" /proc/kallsyms
ffffffffc0a12000 t hello_init    [hello]
ffffffffc0a12050 t hello_exit    [hello]
ffffffffc0a12100 d hello_count   [hello]

# Find where a function lives (e.g., printk)
$ sudo grep " T printk$" /proc/kallsyms
ffffffff81234567 T printk

# Decode an oops address to nearest symbol
$ ADDR=ffffffff81234599; sudo awk -v a="$ADDR" 'strtonum("0x"$1) <= strtonum("0x"a) {s=$0} END {print s}' /proc/kallsyms

# See what the kernel text/data sections look like
$ sudo grep -E "^ffffffff[0-9a-f]+ [A-Z] _(text|etext|sdata|edata|end)$" /proc/kallsyms
ffffffff81000000 T _text      ← kernel code starts here
ffffffff82000000 T _etext     ← kernel code ends here
ffffffff82200000 D _sdata     ← kernel data starts here
ffffffff82600000 B _end       ← kernel ends here

# KASLR note: With KASLR enabled (most systems), addresses in kallsyms
# change every boot. That's why crash addresses must be compared against
# the CURRENT boot's kallsyms, not a saved copy.

Error Return Values — The Kernel's Error Code System

⚠️ How Kernel Functions Signal Errors

Kernel functions return negative errno values on error. These are defined in <linux/errno.h> (values from POSIX). The pattern is always: return 0 on success, return -EXXXX on failure. Never return positive values as errors.

Error CodeValueMeaningWhen to Return from Driver
-EINVAL-22Invalid argumentBad parameter from userspace, invalid ioctl command, out-of-range value
-ENOMEM-12Out of memorykmalloc() returned NULL
-EIO-5I/O errorHardware communication failed, register read/write timed out
-ENODEV-19No such deviceHardware not present, device removed during operation
-EBUSY-16Device or resource busyResource already in use, lock contention, module in use
-EACCES-13Permission deniedCaller lacks required privilege (CAP_SYS_ADMIN, etc.)
-EFAULT-14Bad addresscopy_from/to_user() failed — userspace pointer was invalid
-ETIMEDOUT-110Connection timed outHardware didn't respond within expected time
-EAGAIN-11Try againResource temporarily unavailable — caller should retry
-ENOTSUPP-524Not supportedOperation not supported by this hardware
-EPROBE_DEFER-517Probe deferredDependency not ready yet — kernel will retry probe later. Critical for platform/I2C drivers!
C — PTR_ERR / IS_ERR / ERR_PTR — Encoding Errors in Pointers
#include <linux/err.h>

/* Many kernel functions return either a valid pointer OR an error code
 * encoded as a pointer. This uses the fact that kernel pointers are always
 * above 0xffff... — the very last 4KB of address space is used for errors.
 *
 * IS_ERR(ptr)   → true if ptr is an error (value between -4095 and -1)
 * PTR_ERR(ptr)  → extract the errno from an error pointer (e.g., -ENOMEM)
 * ERR_PTR(err)  → convert an errno to an error pointer */

struct my_device *create_device(const char *name)
{
    struct my_device *dev;

    dev = kzalloc(sizeof(*dev), GFP_KERNEL);
    if (!dev)
        return ERR_PTR(-ENOMEM);  /* return error as pointer */

    return dev;                        /* return valid pointer on success */
}

static int __init my_init(void)
{
    struct my_device *dev;

    dev = create_device("test");
    if (IS_ERR(dev)) {                  /* check if it's an error */
        int err = PTR_ERR(dev);        /* extract the errno: -ENOMEM */
        pr_err("create_device failed: %d\n", err);
        return err;
    }
    /* dev is a valid pointer here */
    pr_info("Device created: %p\n", dev);
    return 0;
}

C — debug_practice.c — Full Practice Module
/*
 * debug_practice.c - Day 3 Practice Module
 * Demonstrates: pr_debug, BUG_ON, WARN_ON, error handling, coding style
 *
 * Build: make
 * Load:  sudo insmod debug_practice.ko
 *        sudo insmod debug_practice.ko mode=crash    (deliberate crash)
 *        sudo insmod debug_practice.ko mode=warn     (WARN_ON demo)
 */
#include <linux/init.h>
#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/slab.h>    /* kmalloc, kfree */
#include <linux/bug.h>     /* BUG_ON, WARN_ON */

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Your Name");
MODULE_DESCRIPTION("Day 3 Debug Practice Module");

static char *mode = "normal";
module_param(mode, charp, 0444);
MODULE_PARM_DESC(mode, "Mode: normal|crash|warn|defer");

struct sensor_data {
    int   value;
    char *name;     /* field at offset 0x08 (after 4-byte value + 4-byte padding) */
};

static int process_sensor(struct sensor_data *s)
{
    pr_debug("process_sensor: ptr=%p\n", s);

    /* WARN_ON: sensor pointer should never be NULL here */
    if (WARN_ON(!s))
        return -EINVAL;

    pr_debug("process_sensor: value=%d, name=%s\n", s->value, s->name);
    pr_info("Sensor '%s' reads: %d\n", s->name, s->value);
    return 0;
}

static int __init debug_practice_init(void)
{
    struct sensor_data *sensor;
    int ret;

    pr_info("debug_practice: Loading in '%s' mode\n", mode);

    if (strcmp(mode, "crash") == 0) {
        /* DELIBERATE NULL DEREFERENCE — run in QEMU! */
        pr_warn("About to crash deliberately! Watch dmesg.\n");
        sensor = NULL;
        pr_info("Sensor name: %s\n", sensor->name);  /* CRASH HERE */
        return 0;
    }

    if (strcmp(mode, "warn") == 0) {
        /* WARN_ON demo — system survives */
        ret = process_sensor(NULL);   /* passes NULL → WARN_ON fires */
        pr_info("process_sensor returned: %d (system still alive!)\n", ret);
        return 0;
    }

    /* NORMAL mode — proper resource management with goto cleanup */
    sensor = kzalloc(sizeof(*sensor), GFP_KERNEL);
    if (unlikely(!sensor)) {
        pr_err("Failed to allocate sensor struct\n");
        ret = -ENOMEM;
        goto err_alloc;
    }

    sensor->value = 42;
    sensor->name = "temp_sensor_0";

    ret = process_sensor(sensor);
    if (unlikely(ret < 0)) {
        pr_err("process_sensor failed: %d\n", ret);
        goto err_process;
    }

    pr_info("debug_practice: Init complete\n");
    kfree(sensor);  /* no persistent state needed */
    return 0;

err_process:
    kfree(sensor);
err_alloc:
    return ret;
}

static void __exit debug_practice_exit(void)
{
    pr_info("debug_practice: Unloaded\n");
}

module_init(debug_practice_init);
module_exit(debug_practice_exit);
