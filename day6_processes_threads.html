struct task_struct — The Process Descriptor

🧬 Every Runnable Entity is a task_struct

In Linux, processes and threads are both represented by struct task_struct — the difference is just which resources they share. A process has its own memory space (mm_struct), file table, and signal table. Threads within the same process share all of these. Even kernel threads are task_structs, but they have mm = NULL (no user address space).

task_struct is defined in include/linux/sched.h and is one of the largest structs in the kernel — over 700 fields across hundreds of lines. You'll interact with a carefully chosen subset as a driver developer.

struct task_struct — Fields Relevant to Driver Developers ────────────────────────────────────────────────────────────────────────────── include/linux/sched.h Identity: pid_t pid; ← Process ID (unique per thread) pid_t tgid; ← Thread Group ID (= pid of main thread) char comm[16]; ← Command name (truncated to 15 chars) struct task_struct *parent; ← Parent process struct task_struct *real_parent; State: unsigned int __state; ← Current execution state (TASK_RUNNING etc.) int exit_state; ← ZOMBIE, DEAD int exit_code; ← exit() argument / signal number Scheduling: int prio; ← Dynamic priority (modified by scheduler) int static_prio; ← Base priority (from nice value) int rt_priority; ← Realtime priority (0 for normal tasks) unsigned int policy; ← SCHED_NORMAL, SCHED_FIFO, SCHED_RR struct sched_entity se; ← CFS scheduler entity (vruntime lives here) Memory: struct mm_struct *mm; ← User address space (NULL for kernel threads) struct mm_struct *active_mm; ← Active mm (borrowed for kernel threads) Files & I/O: struct files_struct *files; ← Open file table struct fs_struct *fs; ← Filesystem context (cwd, root) Signals: struct signal_struct *signal; struct sighand_struct *sighand; sigset_t blocked; ← Blocked signal mask Credentials: const struct cred *cred; ← UID, GID, capabilities const struct cred *real_cred; Waiting / synchronisation: wait_queue_head_t *io_wait; ← Waiting for I/O struct completion *vfork_done; Tracing / debugging: unsigned long ptrace; ← ptrace flags struct thread_struct thread; ← CPU-specific state (registers, FPU)
C — Reading task_struct Fields from a Module
#include <linux/sched.h>       /* task_struct, current */
#include <linux/cred.h>        /* current_uid(), current_euid() */
#include <linux/mm.h>          /* mm_struct */
#include <linux/pid.h>         /* task_pid_nr() */

static void print_current_task_info(void)
{
    /* 'current' is a per-CPU pointer to the running task_struct.
     * It is ONLY valid in process context (syscall, ioctl, etc.)
     * NEVER dereference current in interrupt context — it's meaningless there. */
    struct task_struct *task = current;

    pr_info("=== Current Task Info ===\n");
    pr_info("  comm    = %s\n",  task->comm);         /* process name */
    pr_info("  pid     = %d\n",  task->pid);          /* thread ID */
    pr_info("  tgid    = %d\n",  task->tgid);         /* process ID */
    pr_info("  state   = %u\n",  task->__state);      /* TASK_RUNNING etc. */
    pr_info("  prio    = %d\n",  task->prio);         /* scheduler priority */
    pr_info("  policy  = %u\n",  task->policy);       /* sched policy */
    pr_info("  has mm  = %s\n",  task->mm ? "yes (user)" : "no (kernel thread)");

    /* Safe helpers for credentials (never access cred fields directly) */
    pr_info("  uid     = %u\n",  from_kuid(&init_user_ns, current_uid()));
    pr_info("  euid    = %u\n",  from_kuid(&init_user_ns, current_euid()));

    /* Is caller privileged? (important for security-sensitive ioctls) */
    if (capable(CAP_SYS_ADMIN))
        pr_info("  caller has CAP_SYS_ADMIN\n");

    /* Walk parent chain */
    struct task_struct *p = current;
    pr_info("  Parent chain: ");
    while (p->pid != 0) {
        pr_cont("%s(%d) → ", p->comm, p->pid);
        p = p->real_parent;
    }
    pr_cont("[idle]\n");
}
⚠️
current is Only Valid in Process ContextThe current macro reads a per-CPU variable pointing to the currently executing task. In interrupt context (IRQ handlers, softirqs, tasklets), current points to whatever task happened to be running when the interrupt fired — it has nothing to do with your driver's user. Accessing current in interrupt context leads to subtle, hard-to-reproduce bugs.

Process States & Transitions

📊 Every State a Task Can Be In

The __state field of task_struct tracks the task's current lifecycle state. As a driver developer, you directly control state transitions when you put tasks to sleep waiting for hardware events and wake them when the hardware responds.

State ConstantValueMeaningDriver Relevance
TASK_RUNNING0x0000 On CPU right now or on the run queue waiting for a CPU Your ioctl() handler runs in this state. current->__state == TASK_RUNNING while inside your driver code.
TASK_INTERRUPTIBLE0x0001 Sleeping, waiting for an event. Will be woken by wake_up_interruptible() OR by any signal. Most common sleep state in drivers. Used by wait_event_interruptible(). If the user Ctrl+C's, the task wakes with -ERESTARTSYS.
TASK_UNINTERRUPTIBLE0x0002 Deep sleep — only woken by the specific event, not by signals. Appears as "D" in ps. Used for critical waits that must not be interrupted (disk I/O, NFS). Dangerous if the event never comes — process becomes unkillable.
TASK_KILLABLE0x0102 Like UNINTERRUPTIBLE but can be killed by fatal signals (SIGKILL). Preferred over TASK_UNINTERRUPTIBLE for most driver waits. Use wait_event_killable().
TASK_STOPPED0x0004 Stopped by SIGSTOP or SIGTSTP. Resumed by SIGCONT. Handled by signal infrastructure, not by drivers directly.
TASK_TRACED0x0008 Being traced by ptrace (gdb, strace). Transparent to drivers. Your code runs normally; ptrace intercepts syscall boundaries.

Kernel Threads — kthread API

🧵 Running Background Work in Kernel Space

A kernel thread is a task that runs entirely in kernel space — it has no user address space (mm == NULL), no file descriptors, no signal handlers. Kernel threads are used for background tasks like polling hardware, flushing write-back cache, managing devices, and running periodic maintenance tasks.

Common examples you've seen: kworker/0:0 (workqueue), ksoftirqd/0 (softirq), kswapd0 (memory reclaim), jbd2/sda1-8 (journaling). You'll create your own for devices that need continuous polling or background I/O.

C — Kernel Thread Lifecycle — Full Pattern
#include <linux/kthread.h>     /* kthread_run, kthread_stop, kthread_should_stop */
#include <linux/delay.h>       /* msleep */
#include <linux/sched.h>

/* Thread handle — save this to stop the thread later */
static struct task_struct *poll_thread;

/* ── STEP 1: Write the thread function ────────────────────────────── */
static int sensor_poll_thread(void *data)
{
    struct my_device *dev = (struct my_device *)data;

    pr_info("sensor_poll_thread: started, polling every 100ms\n");

    /* kthread_should_stop() returns true when kthread_stop() is called.
     * Your thread MUST check this and return when it's true.
     * Failure to do so: kthread_stop() will hang forever. */
    while (!kthread_should_stop()) {
        int value;

        /* Do the real work */
        value = read_sensor_register(dev);
        pr_debug("sensor value: %d\n", value);

        if (value > 90)
            pr_warn("sensor: temperature critical: %d°C!\n", value);

        /* msleep_interruptible: sleep for 100ms.
         * Returns 0 on normal wakeup, remaining ms if a signal arrived.
         * This allows the thread to respond to kthread_stop() promptly. */
        msleep_interruptible(100);
    }

    pr_info("sensor_poll_thread: exiting\n");
    return 0;   /* thread exit value — retrievable via kthread_stop() */
}

/* ── STEP 2: Start the thread in probe/init ───────────────────────── */
static int __init driver_init(void)
{
    /* kthread_run(fn, data, namefmt, ...)
     *   fn      = thread function
     *   data    = void* passed to fn (your device struct)
     *   namefmt = printf-style name (shown in ps, top)
     * Returns: task_struct* on success, ERR_PTR on failure */
    poll_thread = kthread_run(sensor_poll_thread, dev, "sensor_poll/%d", dev->id);
    if (IS_ERR(poll_thread)) {
        pr_err("Failed to create poll thread: %ld\n", PTR_ERR(poll_thread));
        poll_thread = NULL;
        return PTR_ERR(poll_thread);
    }
    pr_info("Poll thread started: pid=%d\n", poll_thread->pid);
    return 0;
}

/* ── STEP 3: Stop the thread in remove/exit ───────────────────────── */
static void __exit driver_exit(void)
{
    if (poll_thread) {
        /* kthread_stop() signals the thread to stop (sets a flag that
         * kthread_should_stop() returns true), then WAITS for the thread
         * to actually exit and return. Returns the thread's return value.
         * Never call kthread_stop() after the thread has already exited! */
        int ret = kthread_stop(poll_thread);
        pr_info("Poll thread stopped, return value: %d\n", ret);
    }
}

/* ── ADVANCED: Create without immediately running ─────────────────── */
struct task_struct *t = kthread_create(my_thread_fn, data, "mythread");
if (IS_ERR(t)) return PTR_ERR(t);

/* Set scheduling policy before starting */
struct sched_param param = { .sched_priority = 50 };
sched_setscheduler(t, SCHED_FIFO, &param);   /* realtime! */

/* Pin thread to specific CPU */
kthread_bind(t, 0);   /* bind to CPU 0 */

wake_up_process(t);    /* now start it */
💡
kthread_run vs kthread_createUse kthread_run() when you want to create and immediately start a thread (99% of cases). Use kthread_create() when you need to configure the thread (CPU affinity, scheduling policy) before it starts running. kthread_run(fn, data, name) is just kthread_create() + wake_up_process().

Workqueues — Deferring Work to Process Context

⚙️ The #1 Way to Defer Work from Interrupt Context

A workqueue is a kernel mechanism that lets you schedule a function to be called later, in process context (a kernel thread). This is the standard solution for the classic driver problem: your interrupt handler fires (atomic context — can't sleep, can't do I/O), but the work you need to do requires sleeping or calling kernel APIs that can sleep.

The pattern is: IRQ handler does minimal work (read hardware, clear interrupt), queues a work item, returns. The kernel's worker thread picks up the work item and runs it at a safe time in process context.

IRQ fires
→
irq_handler() — atomic context
Read HW reg, clear IRQ, copy data
→
schedule_work(&dev->work)
Queue the work item
Later...
→
kworker/0:1 — process context
Runs my_work_fn(), can sleep
→
I2C read, wake_up(), notify userspace
C — Workqueue API — Three Patterns
#include <linux/workqueue.h>

/* ═══════════════════════════════════════════════════════════
 * PATTERN 1: System Workqueue (simplest — use for most drivers)
 * The kernel provides a shared workqueue. Schedule work onto it.
 * ═══════════════════════════════════════════════════════════ */

struct my_device {
    struct work_struct  work;      /* embed work_struct in your device struct */
    struct work_struct  work2;
    struct delayed_work dwork;     /* for delayed scheduling */
    u32                 last_value;
};

/* Work function — runs in process context (kworker thread) */
static void my_work_fn(struct work_struct *work)
{
    /* container_of to get back to your device struct */
    struct my_device *dev = container_of(work, struct my_device, work);

    pr_info("Work running in: %s (pid %d)\n", current->comm, current->pid);
    /* We're in process context — CAN sleep, do I2C, allocate with GFP_KERNEL */

    do_slow_i2c_operation(dev);       /* safe: can sleep */
    wake_up(&dev->wait_queue);        /* wake any waiting readers */
}

/* Delayed work function — same signature, runs after a delay */
static void my_delayed_work_fn(struct work_struct *work)
{
    struct my_device *dev =
        container_of(to_delayed_work(work), struct my_device, dwork);
    pr_info("Delayed work fired!\n");
    /* Re-schedule itself for periodic work: */
    schedule_delayed_work(&dev->dwork, msecs_to_jiffies(1000));
}

/* In your probe/init: initialise the work items */
INIT_WORK(&dev->work, my_work_fn);               /* bind function to work item */
INIT_DELAYED_WORK(&dev->dwork, my_delayed_work_fn);

/* In your IRQ handler (atomic context) — queue the work: */
schedule_work(&dev->work);                        /* queue on system_wq */
schedule_delayed_work(&dev->dwork,
                       msecs_to_jiffies(500));    /* run 500ms from now */

/* In your remove/exit: cancel pending work before freeing device */
cancel_work_sync(&dev->work);      /* cancel + wait for completion if running */
cancel_delayed_work_sync(&dev->dwork);

/* ═══════════════════════════════════════════════════════════
 * PATTERN 2: Dedicated Workqueue (for work that needs isolation)
 * ═══════════════════════════════════════════════════════════ */

/* Create a dedicated single-threaded workqueue for your driver */
struct workqueue_struct *my_wq = alloc_ordered_workqueue("my_driver_wq", 0);
/* alloc_ordered_workqueue: single thread, work items run sequentially
 * alloc_workqueue: multi-thread, work may run concurrently              */

/* Queue work onto YOUR workqueue (not the system one) */
queue_work(my_wq, &dev->work);
queue_delayed_work(my_wq, &dev->dwork, msecs_to_jiffies(100));

/* In exit: flush (wait for all pending work to finish) then destroy */
flush_workqueue(my_wq);    /* wait for ALL pending items to complete */
destroy_workqueue(my_wq);

/* ═══════════════════════════════════════════════════════════
 * PATTERN 3: Check if work is pending / running
 * ═══════════════════════════════════════════════════════════ */
if (work_pending(&dev->work))
    pr_info("Work is queued but hasn't run yet\n");

if (work_busy(&dev->work))
    pr_info("Work is queued or currently running\n");

📋 System Workqueues

The kernel provides pre-made workqueues for common use. schedule_work() uses system_wq by default. Others:

  • system_wq — general purpose, concurrent
  • system_highpri_wq — high priority work
  • system_long_wq — long-running work (won't starve others)
  • system_freezable_wq — frozen during suspend
  • system_power_efficient_wq — batches work for power saving

⚡ Tasklets vs Workqueues

Tasklets (an older mechanism) run in softirq / atomic context — no sleeping allowed. Workqueues run in process context — sleeping allowed.

  • Need to sleep? → Workqueue
  • Need lowest latency, no sleep? → Tasklet (being deprecated)
  • Modern drivers: use workqueues for almost everything
  • Tasklets are marked deprecated in kernel 5.17+ — new drivers should use workqueues or threaded IRQs instead

Wait Queues — Sleeping Until an Event Occurs

😴 The Core Sleep/Wake Mechanism in Drivers

Wait queues are how drivers implement blocking I/O. When a user process calls read() on your device and no data is available, you put it to sleep on a wait queue. When hardware generates an interrupt and data arrives, your IRQ handler wakes the wait queue. The sleeping process is then scheduled to run again and completes the read.

This is the foundation of every blocking device driver read, write, poll, and select operation. Get this right and the rest of blocking I/O falls into place.

C — Wait Queue — Complete Sleep/Wake Pattern
#include <linux/wait.h>
#include <linux/sched.h>

struct my_device {
    wait_queue_head_t read_wq;    /* processes waiting to read */
    bool              data_ready; /* the condition they're waiting for */
    u8                data[64];
    int               data_len;
};

/* ── INIT: in probe() ─────────────────────────────────────────────── */
init_waitqueue_head(&dev->read_wq);    /* dynamic init */
/* OR static:  DECLARE_WAIT_QUEUE_HEAD(my_wq); */

/* ── CONSUMER: read() syscall handler — process context ──────────── */
static ssize_t my_read(struct file *filp, char __user *buf,
                        size_t count, loff_t *ppos)
{
    struct my_device *dev = filp->private_data;
    int ret;

    /* wait_event_interruptible(wq, condition):
     *   Puts this task to SLEEP (TASK_INTERRUPTIBLE) until:
     *     1. condition becomes TRUE  → wakes normally, returns 0
     *     2. A signal arrives        → wakes, returns -ERESTARTSYS
     *
     * The condition is evaluated ATOMICALLY before sleeping —
     * no missed-wakeup race with the IRQ handler. */
    ret = wait_event_interruptible(dev->read_wq, dev->data_ready);
    if (ret == -ERESTARTSYS)
        return -EINTR;   /* user pressed Ctrl+C — interrupted by signal */

    /* Data is ready — copy to userspace */
    int to_copy = min((size_t)dev->data_len, count);
    if (copy_to_user(buf, dev->data, to_copy))
        return -EFAULT;

    dev->data_ready = false;   /* reset for next read */
    return to_copy;
}

/* ── PRODUCER: IRQ handler — atomic context ─────────────────────── */
static irqreturn_t my_irq_handler(int irq, void *dev_id)
{
    struct my_device *dev = dev_id;

    /* Read data from hardware into buffer */
    dev->data_len = read_hw_fifo(dev->data, sizeof(dev->data));

    /* Set the condition TRUE before waking — order matters! */
    dev->data_ready = true;

    /* wake_up_interruptible: wake one task waiting on read_wq
     * The task's wait_event_interruptible() will recheck the condition. */
    wake_up_interruptible(&dev->read_wq);

    return IRQ_HANDLED;
}

/* ── ALL wait_event VARIANTS ─────────────────────────────────────── */
/*
 * wait_event(wq, cond)
 *   Uninterruptible. Bad: process becomes unkillable if cond never true.
 *
 * wait_event_interruptible(wq, cond)             returns -ERESTARTSYS on signal
 *   Most common. Woken by signal or event.
 *
 * wait_event_killable(wq, cond)                  returns -ERESTARTSYS on SIGKILL
 *   Like interruptible but only SIGKILL wakes it.
 *
 * wait_event_timeout(wq, cond, timeout)          returns remaining jiffies or 0
 *   Give up after timeout jiffies even if cond not true.
 *
 * wait_event_interruptible_timeout(wq, cond, timeout)
 *   Timeout + interruptible. Returns 0=timeout, >0=ok, -ERESTARTSYS=signal.
 *   This is the one to use for most device reads with a timeout.
 *
 * wait_event_lock_irq(wq, cond, lock)
 *   Releases a spinlock while sleeping, reacquires before returning.
 */
🚨
The Missed Wakeup Race — Why Conditions MatterNever write: if (!data_ready) { sleep_on(&wq); } — there's a race: the IRQ could fire between the check and the sleep, and you'd sleep forever. wait_event_interruptible(wq, cond) solves this by atomically checking the condition and going to sleep, ensuring no wakeup can be missed. Always use the wait_event* family — never sleep_on() (deprecated).

Completions — One-Shot Synchronisation

✅ Simpler than Wait Queues for "Done" Signalling

A completion is a lightweight, one-shot version of a wait queue. Where wait queues are general-purpose (multiple waiters, re-usable condition), a completion is specifically designed for the pattern: "Thread A does some work, then signals Thread B that the work is done." Completions are simpler to use correctly and are used throughout the kernel for initialisation sequences, device I/O completion, and thread synchronisation.

C — Completions — Declare, Wait, Complete
#include <linux/completion.h>

/* ── DECLARE & INIT ───────────────────────────────────────────────── */
static DECLARE_COMPLETION(firmware_loaded);   /* static — starts "not done" */

/* Dynamic: */
struct completion cmd_done;
init_completion(&cmd_done);   /* runtime init — also "not done" */

/* ── WAIT for completion (blocks the calling thread) ────────────────── */
wait_for_completion(&firmware_loaded);
/* Blocks until complete() is called. Uninterruptible. */

wait_for_completion_interruptible(&firmware_loaded);
/* Like above but returns -ERESTARTSYS if a signal arrives */

unsigned long ret = wait_for_completion_timeout(&cmd_done,
                        msecs_to_jiffies(5000));
if (ret == 0) {
    pr_err("Command timed out after 5 seconds!\n");
    return -ETIMEDOUT;
}
/* ret > 0: completed in time (ret = remaining jiffies) */

/* ── SIGNAL completion (wakes the waiter) ─────────────────────────── */
complete(&cmd_done);          /* wake ONE waiter */
complete_all(&cmd_done);      /* wake ALL waiters (permanently done) */

/* ── REAL EXAMPLE: Command/response pattern ──────────────────────── */
struct my_cmd {
    u8                 opcode;
    u8                 result;
    struct completion done;     /* per-command completion */
};

static int send_command(struct my_device *dev, u8 opcode)
{
    struct my_cmd cmd = {};
    cmd.opcode = opcode;
    init_completion(&cmd.done);

    /* Send command to hardware */
    dev->pending_cmd = &cmd;
    write_hw_register(dev, REG_CMD, opcode);

    /* Wait up to 1 second for hardware to respond */
    if (!wait_for_completion_timeout(&cmd.done, msecs_to_jiffies(1000)))
        return -ETIMEDOUT;

    return (cmd.result == 0) ? 0 : -EIO;
}

/* IRQ handler — called when hardware sends response */
static irqreturn_t cmd_irq_handler(int irq, void *dev_id)
{
    struct my_device *dev = dev_id;
    dev->pending_cmd->result = read_hw_register(dev, REG_RESULT);
    complete(&dev->pending_cmd->done);    /* wake the send_command() caller */
    return IRQ_HANDLED;
}

Kernel Timing — Delays, Sleeps & Jiffies

⏱️ Timing in the Kernel — Every Option Explained

The kernel provides multiple timing mechanisms, and choosing the wrong one is a very common mistake. The core issue: busy-wait delays burn CPU cycles (nothing else runs), while sleep-based delays yield the CPU but have minimum latency overhead. Know when to use each.

FunctionRangeAccuracyBlocks CPU?ContextUse When
ndelay(ns) 1–1000 ns Hardware dependent YES — busy-wait Any (incl. IRQ) Tiny hardware timing gaps (register write delays). Very short only.
udelay(us) 1–1000 µs Hardware dependent YES — busy-wait Any (incl. IRQ) Hardware setup/hold times, SPI bit-bang delays. Keep <100µs.
mdelay(ms) 1–∞ ms Exact (busy-wait) YES — busy-wait Any (incl. IRQ) Avoid — use msleep() instead. Only justified in atomic context.
msleep(ms) ≥1 ms ±1 jiffy (~10ms) No — yields CPU Process only Most common for driver delays. Hardware power-on waits, probe delays.
msleep_interruptible(ms) ≥1 ms ±1 jiffy No — yields CPU Process only Sleep that can be cut short by signals. Use in kthread loops.
usleep_range(min,max) 10–20000 µs min–max range No — yields CPU Process only Sub-millisecond sleeps where hrtimer is acceptable. Batches wakeups for power efficiency.
schedule_timeout(j) Any jiffies ±1 jiffy No — yields CPU Process only When you need to yield for a number of jiffies and handle task state yourself.
C — Timing Conversions & Jiffies
#include <linux/delay.h>       /* udelay, msleep, usleep_range */
#include <linux/jiffies.h>     /* jiffies, HZ, time_after */
#include <linux/ktime.h>       /* ktime_get, ktime_to_ns */

/* ── JIFFIES — the kernel's clock tick counter ───────────────────── */
/* HZ = ticks per second. Typically 100, 250, or 1000 Hz.
 * jiffies = unsigned long counter, wraps around. Never compare directly! */

/* Convert between jiffies and time units: */
unsigned long j = msecs_to_jiffies(500);    /* 500ms → jiffies */
unsigned long m = jiffies_to_msecs(j);       /* jiffies → ms */
unsigned long u = usecs_to_jiffies(10000);  /* 10000µs → jiffies */
unsigned long s = secs_to_jiffies(5);       /* 5s → jiffies */

/* CORRECT jiffy comparison (handles wrap-around): */
unsigned long deadline = jiffies + msecs_to_jiffies(5000);
if (time_after(jiffies, deadline))
    pr_warn("Deadline passed!\n");

/* Other time comparisons: */
/* time_after(a, b)       = true if a is after  b (a > b, wrap-safe) */
/* time_before(a, b)      = true if a is before b (a < b, wrap-safe) */
/* time_after_eq(a, b)    = true if a >= b                            */
/* time_in_range(a, b, c) = true if b <= a <= c                       */

/* ── HIGH-RESOLUTION TIMING (nanosecond precision) ──────────────── */
ktime_t start = ktime_get();          /* read monotonic clock (ns since boot) */
do_some_work();
ktime_t end = ktime_get();
s64 elapsed_ns = ktime_to_ns(ktime_sub(end, start));
pr_info("Operation took %lld ns\n", elapsed_ns);

/* ktime_get_real(): wall clock (UTC). ktime_get(): monotonic (better for deltas) */

/* ── USAGE EXAMPLES ──────────────────────────────────────────────── */

/* Hardware power-on: chip needs 10ms after reset before responding */
gpio_set_value(dev->reset_gpio, 1);  /* deassert reset */
msleep(10);                           /* wait 10ms for chip to start up */

/* SPI chip-select hold time: 50ns between CS assert and first clock */
gpio_set_value(dev->cs_gpio, 0);     /* assert CS */
ndelay(50);                           /* 50ns setup time (from datasheet) */

/* Polling hardware register with a timeout */
int wait_for_ready(struct my_device *dev)
{
    unsigned long timeout = jiffies + msecs_to_jiffies(100);
    while (!(read_hw_status(dev) & STATUS_READY)) {
        if (time_after(jiffies, timeout))
            return -ETIMEDOUT;
        usleep_range(100, 200);  /* yield CPU between polls */
    }
    return 0;
}

Iterating All Processes — for_each_process

C — Walking the Process Table
#include <linux/sched.h>
#include <linux/sched/signal.h>   /* for_each_process */
#include <linux/rcupdate.h>       /* rcu_read_lock */

/* ── Walk ALL processes (thread group leaders) ────────────────────── */
static void list_all_processes(void)
{
    struct task_struct *task;

    /* MUST hold RCU read lock while iterating — prevents tasks from
     * being freed while we're looking at them. */
    rcu_read_lock();

    for_each_process(task) {
        /* task → thread group leader (one per process) */
        pr_info("pid=%5d tgid=%5d state=%u  %s\n",
                task->pid, task->tgid,
                task->__state, task->comm);
    }

    rcu_read_unlock();
}

/* ── Walk ALL threads in ALL processes ───────────────────────────── */
static void list_all_threads(void)
{
    struct task_struct *process, *thread;

    rcu_read_lock();

    for_each_process(process) {           /* outer: each process */
        for_each_thread(process, thread) { /* inner: each thread in process */
            pr_info("  tid=%d (%s) in process %d (%s)\n",
                    thread->pid, thread->comm,
                    process->pid, process->comm);
        }
    }

    rcu_read_unlock();
}

/* ── Find a process by PID ────────────────────────────────────────── */
static struct task_struct *find_task_by_pid(pid_t target_pid)
{
    struct task_struct *task, *found = NULL;

    rcu_read_lock();
    /* More efficient than for_each_process for single lookup: */
    task = pid_task(find_vpid(target_pid), PIDTYPE_PID);
    if (task)
        pr_info("Found pid %d: %s\n", target_pid, task->comm);
    rcu_read_unlock();

    return task;  /* WARNING: may be freed after rcu_read_unlock! */
    /* Use get_task_struct(task) inside the lock to hold a reference */
}

/*
 * threads_practice.c — Day 6: Processes, Threads & task_struct
 *
 * Demonstrates:
 *   - kthread with kthread_should_stop() loop
 *   - workqueue item scheduled from the thread
 *   - completion for init signalling

#include <linux/init.h>
#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/kthread.h>
#include <linux/workqueue.h>
#include <linux/completion.h>
#include <linux/wait.h>
#include <linux/delay.h>
#include <linux/jiffies.h>
#include <linux/sched.h>
#include <linux/atomic.h>

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Your Name");
MODULE_DESCRIPTION("Day 6: Threads & Work Deferral Practice");

static int  interval_ms = 1000;
static int  iterations  = 5;
module_param(interval_ms, int, 0444);
module_param(iterations,  int, 0444);
MODULE_PARM_DESC(interval_ms, "Poll interval in ms (default: 1000)");
MODULE_PARM_DESC(iterations,  "Number of iterations before stopping (default: 5)");

/* ── Device state ─────────────────────────────────────────────────── */
struct practice_dev {
    struct task_struct  *poll_thread;
    struct work_struct   process_work;
    struct completion    init_done;     /* signals when ready */
    wait_queue_head_t    data_wq;       /* for simulated read() */
    atomic_t             counter;       /* the "sensor" value */
    bool                 data_ready;
    int                  last_value;
};

static struct practice_dev pdev;

/* ── Work handler — runs in kworker process context ──────────────── */
static void process_work_fn(struct work_struct *work)
{
    struct practice_dev *dev =
        container_of(work, struct practice_dev, process_work);

    pr_info("  [work_fn] running in '%s' (pid=%d), value=%d\n",
            current->comm, current->pid, dev->last_value);

    /* Simulate slow processing (we can sleep here!) */
    msleep(50);
    pr_info("  [work_fn] processing complete\n");

    /* Signal waiting readers */
    dev->data_ready = true;
    wake_up_interruptible(&dev->data_wq);
}

/* ── Poll thread — periodically "reads hardware" ─────────────────── */
static int poll_thread_fn(void *data)
{
    struct practice_dev *dev = data;
    int count = 0;

    pr_info("[poll_thread] started: '%s' (pid=%d)\n",
            current->comm, current->pid);

    /* Signal that init is complete */
    complete(&dev->init_done);

    while (!kthread_should_stop()) {
        int val;

        /* Simulate reading a hardware counter */
        val = atomic_inc_return(&dev->counter);
        dev->last_value = val;
        count++;

        pr_info("[poll_thread] iter %d/%d: value=%d\n",
                count, iterations, val);

        /* Defer heavy processing to workqueue */
        schedule_work(&dev->process_work);

        /* Check if we've done enough iterations */
        if (count >= iterations) {
            pr_info("[poll_thread] done %d iterations, exiting\n", count);
            break;
        }

        /* Sleep between polls — responds to kthread_stop() quickly */
        msleep_interruptible(interval_ms);
    }

    pr_info("[poll_thread] exiting\n");
    return 0;
}

static int __init threads_practice_init(void)
{
    int ret;

    pr_info("=== Day 6: Threads Practice (interval=%dms, iters=%d) ===\n",
            interval_ms, iterations);
    pr_info("[init] running as '%s' (pid=%d)\n",
            current->comm, current->pid);

    /* Initialise device state */
    init_completion(&pdev.init_done);
    init_waitqueue_head(&pdev.data_wq);
    INIT_WORK(&pdev.process_work, process_work_fn);
    atomic_set(&pdev.counter, 0);
    pdev.data_ready = false;

    /* Start the polling thread */
    pdev.poll_thread = kthread_run(poll_thread_fn, &pdev, "practice_poll");
    if (IS_ERR(pdev.poll_thread)) {
        ret = PTR_ERR(pdev.poll_thread);
        pdev.poll_thread = NULL;
        pr_err("Failed to create thread: %d\n", ret);
        return ret;
    }

    /* Wait for thread to signal it's ready (timeout: 2s) */
    if (!wait_for_completion_timeout(&pdev.init_done, msecs_to_jiffies(2000))) {
        pr_err("Thread did not start within 2s!\n");
        kthread_stop(pdev.poll_thread);
        return -ETIMEDOUT;
    }

    pr_info("[init] thread started successfully, module ready\n");
    return 0;
}

static void __exit threads_practice_exit(void)
{
    pr_info("[exit] stopping...\n");

    if (pdev.poll_thread)
        kthread_stop(pdev.poll_thread);

    /* Wait for any queued work to finish before module is removed */
    cancel_work_sync(&pdev.process_work);

    pr_info("[exit] done. Final counter: %d\n",
            atomic_read(&pdev.counter));
}

module_init(threads_practice_init);
module_exit(threads_practice_exit);
